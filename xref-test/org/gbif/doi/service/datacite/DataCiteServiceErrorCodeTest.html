25  /**
26   *
27   * Test behavior of DataCiteService when the underlying HttpClient returns Error codes.
28   *
29   */
30  public class DataCiteServiceErrorCodeTest {
31  
32    public final static URI TEST_TARGET = URI.create("http://www.gbif.org/datasets");
33    private File TEST_METADATA = FileUtils.getClasspathFile("metadata/minimal.xml");
34    private DOI TEST_DOI = new DOI(DOI.TEST_PREFIX, "test");
35  
36    private int REQUEST_ENTITY_TOO_LARGE = 413;
37  
38    private DataCiteService buildDataCiteService(){
39      try {
40        ObjectMapper mapper = new ObjectMapper(new YAMLFactory());
41        InputStream dc = FileUtils.classpathStream("datacite.yaml");
42        ServiceConfig cfg = mapper.readValue(dc, ServiceConfig.class);
43        return new DataCiteService(new MockHttpClient(REQUEST_ENTITY_TOO_LARGE), cfg);
44      } catch (IOException e) {
45        throw new IllegalStateException(e);
46      }
47    }
48  
49    @Test
50    public void testRequestEntityTooLarge() throws Exception {
51  
52      boolean doiHttpExceptionCaught = false;
53      DataCiteService service = buildDataCiteService();
54      service = spy(service);
55  
56      // in order to test the HTTP code REQUEST_ENTITY_TOO_LARGE we need to mock the resolve(DOI) method.
57      try {
58        when(service.resolve(TEST_DOI)).thenReturn(new DoiData(DoiStatus.REGISTERED, TEST_TARGET));
59        service.update(TEST_DOI, org.apache.commons.io.FileUtils.readFileToString(TEST_METADATA));
60      }
61      catch (DoiHttpException doiHttpEx){
62        doiHttpExceptionCaught = true;
63        assertEquals(REQUEST_ENTITY_TOO_LARGE, doiHttpEx.getStatus());
64      }
65  
66      assertTrue("When HttpClient triggers a 413, DoiHttpException is thrown ", doiHttpExceptionCaught);
67    }

69  }
