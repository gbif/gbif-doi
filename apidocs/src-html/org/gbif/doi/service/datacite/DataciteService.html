001package org.gbif.doi.service.datacite;
002
003import org.gbif.api.model.common.DOI;
004import org.gbif.doi.service.DoiException;
005import org.gbif.doi.service.DoiExistsException;
006import org.gbif.doi.service.DoiHttpException;
007import org.gbif.doi.metadata.datacite.DataCiteMetadata;
008import org.gbif.doi.service.BaseService;
009import org.gbif.doi.service.DoiStatus;
010import org.gbif.doi.service.ServiceConfig;
011
012import java.net.URI;
013
014import com.google.common.annotations.VisibleForTesting;
015import com.google.common.base.Charsets;
016import com.google.common.base.Preconditions;
017import com.google.common.base.Strings;
018import org.apache.http.HttpEntity;
019import org.apache.http.client.methods.HttpPost;
020import org.apache.http.entity.ContentType;
021import org.apache.http.entity.StringEntity;
022import org.apache.http.impl.client.CloseableHttpClient;
023import org.slf4j.Logger;
024import org.slf4j.LoggerFactory;
025
026/**
027 * This DataCite service implementation.
028 * https://mds.datacite.org/static/apidoc
029 *
030 * Warning! The datacite implementation does not re-activate previously deleted DOIs when calling register() again.
031 * You also need to update the DOI with new metadata to re-activate the DOI!!!
032 */
033public class DataciteService extends BaseService {
034
035  private static final Logger LOG = LoggerFactory.getLogger(DataciteService.class);
036  private static final ContentType APPLICATION_XML_UTF8 = ContentType.create("application/xml", Charsets.UTF_8);
037  private static final URI DEFAULT_API = URI.create("https://mds.datacite.org/");
038  // god knows why, but datacite assigns this URI to all reserved test DOIs as target URLs
039  private static final URI TEST_404_TARGET = URI.create("http://www.datacite.org/testprefix");
040  private final URI apiWs;
041  private final URI doiWs;
042  private final URI metadataWs;
043
044  public DataciteService(CloseableHttpClient httpClient, ServiceConfig cfg) {
045    super(httpClient, cfg);
046    apiWs = cfg.getApi() == null ? DEFAULT_API : cfg.getApi();
047    doiWs = apiWs.resolve("/doi/");
048    metadataWs = apiWs.resolve("/metadata/");
049  }
050
051  /**
052   * Detecting the status might need 2 calls.
053   * A reserved DOI returns metadata and either the test or NULL target URL.
054   * An active registered DOI returns metadata and a target URI
055   * A deleted but registered DOI returns a http 410 for the metadata and still the target URI.
056   */
057  @Override
058  public DoiStatus resolve(DOI doi) throws DoiException {
059    Preconditions.checkNotNull(doi);
060    try {
061
062      final URI target = getTargetUrl(doi);
063      try {
064        get(metadataUri(doi));
065        return target == null ? new DoiStatus(DoiStatus.Status.RESERVED, null) : new DoiStatus(DoiStatus.Status.REGISTERED, target);
066
067      } catch (DoiHttpException e) {
068        // we only see a 410 for the metadata when a registered DOI was deleted
069        if (e.getStatus() == 410) {
070          return new DoiStatus(DoiStatus.Status.DELETED, target);
071        }
072      }
073
074    } catch (DoiException e) {
075      LOG.debug("Unknown error detecting DOI status", e.getMessage());
076    }
077    return null;
078  }
079
080  private URI getTargetUrl(DOI doi) throws DoiHttpException {
081    // get target URI
082    String content = null;
083    try {
084      content = get(doiUri(doi));
085      if (!Strings.isNullOrEmpty(content)) {
086        URI target = URI.create(content);
087        if (!TEST_404_TARGET.equals(target)) {
088          return target;
089        }
090      }
091    } catch (IllegalArgumentException e) {
092      LOG.debug("DataCite returned an invalid target URL {}", content, e.getMessage());
093    } catch (DoiHttpException e) {
094      if (Math.round( e.getStatus() / 100.0) == 4) {
095        throw e;
096      }
097      LOG.debug("Non existing DOI", e.getMessage());
098    } catch (DoiException e) {
099      LOG.debug("Non existing DOI", e.getMessage());
100    }
101    return null;
102  }
103
104  /**
105   * @return true if the DOI is reserved or registered
106   */
107  private boolean exists(DOI doi) throws DoiException {
108    try {
109      get(metadataUri(doi));
110    } catch (DoiHttpException e) {
111      return false;
112    }
113    return true;
114  }
115
116  @Override
117  public void reserve(DOI doi, DataCiteMetadata metadata) throws DoiException {
118    Preconditions.checkNotNull(doi);
119    Preconditions.checkNotNull(metadata);
120    if (exists(doi)) {
121      throw new DoiExistsException(doi);
122    }
123    post(doi, metadataWs, metadata);
124    LOG.info("Reserved {}", doi);
125  }
126
127  @Override
128  public void register(DOI doi, URI target) throws DoiException {
129    Preconditions.checkNotNull(doi);
130    Preconditions.checkNotNull(target);
131    DoiStatus status = resolve(doi);
132    if (status == null) {
133      throw new DoiException("DOI does not exist");
134    }
135    if (DoiStatus.Status.REGISTERED == status.getStatus()) {
136      throw new DoiExistsException(doi);
137    }
138    authCall(buildDoiUrlPost(doi, target));
139    LOG.info("Registered {}", doi);
140  }
141
142  @Override
143  public boolean delete(DOI doi) throws DoiException {
144    Preconditions.checkNotNull(doi);
145    delete(metadataUri(doi));
146    LOG.info("Deleted {}", doi);
147    return resolve(doi) != null;
148  }
149
150  @Override
151  public void update(DOI doi, DataCiteMetadata metadata) throws DoiException {
152    Preconditions.checkNotNull(doi);
153    Preconditions.checkNotNull(metadata);
154    post(doi, metadataWs, metadata);
155    LOG.info("Updated metadata for {}", doi);
156  }
157
158  @Override
159  public void update(DOI doi, URI target) throws DoiException {
160    Preconditions.checkNotNull(doi);
161    Preconditions.checkNotNull(target);
162
163    DoiStatus status = resolve(doi);
164    if (status == null || DoiStatus.Status.REGISTERED != status.getStatus()) {
165      throw new DoiException("DOI was not registered yet");
166    }
167    authCall(buildDoiUrlPost(doi, target));
168    LOG.info("Updated target URL for {}", doi);
169  }
170
171  private HttpPost buildDoiUrlPost(DOI doi, URI target) {
172    HttpPost post = new HttpPost(doiWs);
173    String body = String.format("doi=%s\nurl=%s", doi.getDoiName(), target);
174    HttpEntity entity = new StringEntity(body, PLAIN_TEXT_UTF8);
175    post.setEntity(entity);
176    return post;
177  }
178
179  @VisibleForTesting
180  protected URI doiUri(DOI doi) throws DoiException {
181    try {
182      return URI.create(doiWs.toString() + doi.getDoiName());
183    } catch (IllegalArgumentException e) {
184      throw new DoiException("Unable to convert DOI to DataCite URI", e);
185    }
186  }
187
188  @VisibleForTesting
189  protected  URI metadataUri(DOI doi) throws DoiException {
190    try {
191      return URI.create(metadataWs.toString() + doi.getDoiName());
192    } catch (IllegalArgumentException e) {
193      throw new DoiException("Unable to convert DOI to DataCite URI", e);
194    }
195  }
196
197  /**
198   * Executes an HTTP POST request.
199   *
200   * @param uri  to issue request against
201   * @param data request body content to be serialized as XML with jackson
202   *
203   * @throws org.gbif.doi.service.DoiException in case anything goes wrong during the request.
204   */
205  private void post(DOI doi, URI uri, DataCiteMetadata data) throws DoiException {
206    LOG.debug("POST: {}", uri);
207    HttpPost req = new HttpPost(uri);
208    // body
209    if (data != null) {
210      try {
211        String xml = toXml(doi, data);
212        HttpEntity entity = new StringEntity(xml, APPLICATION_XML_UTF8);
213        req.setEntity(entity);
214        authCall(req);
215
216      } catch (Exception e) {
217        throw new DoiException(e);
218      }
219    }
220  }
