001package org.gbif.doi.service.datacite;
002
003import org.gbif.api.model.common.DOI;
004import org.gbif.doi.metadata.datacite.DataCiteMetadata;
005import org.gbif.doi.service.InvalidMetadataException;
006
007import java.io.ByteArrayInputStream;
008import java.io.IOException;
009import java.io.InputStream;
010import java.io.StringReader;
011import java.io.StringWriter;
012import javax.xml.bind.JAXBContext;
013import javax.xml.bind.JAXBException;
014import javax.xml.bind.Marshaller;
015import javax.xml.bind.Unmarshaller;
016import javax.xml.transform.stream.StreamSource;
017import javax.xml.validation.Schema;
018import javax.xml.validation.SchemaFactory;
019import javax.xml.validation.Validator;
020
021import com.google.common.base.Charsets;
022import org.slf4j.Logger;
023import org.slf4j.LoggerFactory;
024import org.xml.sax.SAXException;
025
026/**
027 * For the validation of DataCite metadata files.
028 */
029public class DataCiteValidator {
030  private static final Logger LOG = LoggerFactory.getLogger(DataCiteValidator.class);
031  private static final String DATACITE_SCHEMA_LOCATION = "http://datacite.org/schema/kernel-3 http://schema.datacite.org/meta/kernel-3/metadata.xsd";
032  private static Validator DATACITE_VALIDATOR;
033  private static final JAXBContext context;
034  static {
035    try {
036      context = JAXBContext.newInstance(DataCiteMetadata.class);
037    } catch (JAXBException e) {
038      throw new IllegalStateException("Fail to setup JAXB", e);
039    }
040  }
041
042
043  /**
044   * Produces a validated xml representation of the data cite metadata.
045   * The DOI inside the metadata is overwritten to always represent the given DOI
046   * and the generated XML is validated according to the datacite xsd schema.
047   * @return xml string
048   * @throws InvalidMetadataException if the generated xml failed validation, e.g. missing mandatory elements
049   */
050  public static String toXml(DOI doi, DataCiteMetadata data) throws InvalidMetadataException {
051    // make sure we use the right doi inside the metadata
052    data.setIdentifier(DataCiteMetadata.Identifier.builder()
053        .withValue(doi.getDoiName())
054        .withIdentifierType("DOI")
055        .build()
056    );
057    // (un)marshaller are not thread safe and need to be created on each authCall
058    StringWriter writer = new StringWriter();
059    try {
060      Marshaller m = context.createMarshaller();
061      m.setProperty(Marshaller.JAXB_FORMATTED_OUTPUT, Boolean.TRUE);
062      m.setProperty(Marshaller.JAXB_SCHEMA_LOCATION, DATACITE_SCHEMA_LOCATION);
063
064      m.marshal(data, writer);
065      writer.flush();
066    } catch (JAXBException e) {
067      throw new InvalidMetadataException(doi, e);
068    }
069
070    final String xml = writer.toString();
071    // validate the xml before we return it
072    try {
073      validateMetadata(xml);
074      LOG.debug("Metadata XML passed validation", doi);
075
076    } catch (SAXException e) {
077      throw new InvalidMetadataException(doi, e);
078
079    } catch (IOException e) {
080      // cant happen, we had a string in memory already!
081      throw new IllegalStateException(e);
082    }
083    return xml;
084  }
085
086  public static DataCiteMetadata fromXml(String xml) throws JAXBException {
087    return fromXml(new ByteArrayInputStream(xml.getBytes(Charsets.UTF_8)));
088  }
089
090  public static DataCiteMetadata fromXml(InputStream in) throws JAXBException {
091    // (un)marshaller are not thread safe and need to be created on each call
092    Unmarshaller m = context.createUnmarshaller();
093    return (DataCiteMetadata) m.unmarshal(in);
094  }
095
096  /**
097   * @return an xml validator based on the DataCite metadata schema version 3
098   */
099  public static void validateMetadata(String xml) throws IOException, SAXException {
100    getValidator().validate(new StreamSource(new StringReader(xml)));
101  }
102
103  public static void validateMetadata(InputStream xml) throws IOException, SAXException {
104    getValidator().validate(new StreamSource(xml));
105  }
106
107  private static Validator getValidator() throws IOException, SAXException {
108    if (DATACITE_VALIDATOR == null) {
109      // define the type of schema - we use W3C:
110      String schemaLang = "http://www.w3.org/2001/XMLSchema";
111      // resolve validation driver:
112      SchemaFactory factory = SchemaFactory.newInstance(schemaLang);
113      // create schema by reading it from an URL:
114      //File f = FileUtils.getClasspathFile("datacite/metadata.xsd");
115      Schema schema = factory.newSchema(new StreamSource("http://schema.datacite.org/meta/kernel-3/metadata.xsd"));
116      DATACITE_VALIDATOR = schema.newValidator();
117    }
118    return DATACITE_VALIDATOR;
119  }
120
121}
