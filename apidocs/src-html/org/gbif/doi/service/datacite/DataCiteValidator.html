001package org.gbif.doi.service.datacite;
002
003import org.gbif.api.model.common.DOI;
004import org.gbif.doi.metadata.datacite.DataCiteMetadata;
005import org.gbif.doi.service.exception.InvalidMetadataException;
006
007import java.io.ByteArrayInputStream;
008import java.io.IOException;
009import java.io.InputStream;
010import java.io.StringReader;
011import java.io.StringWriter;
012import javax.xml.bind.JAXBContext;
013import javax.xml.bind.JAXBException;
014import javax.xml.bind.Marshaller;
015import javax.xml.bind.Unmarshaller;
016import javax.xml.transform.Source;
017import javax.xml.transform.stream.StreamSource;
018import javax.xml.validation.Schema;
019import javax.xml.validation.SchemaFactory;
020import javax.xml.validation.Validator;
021
022import com.google.common.base.Charsets;
023import org.slf4j.Logger;
024import org.slf4j.LoggerFactory;
025import org.xml.sax.SAXException;
026
027/**
028 * For the validation of DataCite metadata files.
029 */
030public final class DataCiteValidator {
031  private static final Logger LOG = LoggerFactory.getLogger(DataCiteValidator.class);
032  private static final String DATACITE_XSD_LOCATION = "http://schema.datacite.org/meta/kernel-4/metadata.xsd";
033  private static final String DATACITE_SCHEMA_LOCATION = "http://datacite.org/schema/kernel-4 " + DATACITE_XSD_LOCATION;
034  //lazy initialized
035  private static Schema dataciteSchema;
036  private static final JAXBContext CONTEXT;
037
038  private DataCiteValidator() {
039  }
040
041  static {
042    try {
043      CONTEXT = JAXBContext.newInstance(DataCiteMetadata.class);
044    } catch (JAXBException e) {
045      throw new IllegalStateException("Fail to setup JAXB", e);
046    }
047  }
048
049  /**
050   * Produces a validated xml representation of the data cite metadata.
051   * The DOI inside the metadata is overwritten to always represent the given DOI
052   * and the generated XML is validated according to the datacite xsd schema.
053   *
054   * @return xml string
055   * @throws InvalidMetadataException if the generated xml failed validation, e.g. missing mandatory elements
056   */
057  public static String toXml(DOI doi, DataCiteMetadata data) throws InvalidMetadataException {
058    // make sure we use the right doi inside the metadata
059    data.setIdentifier(DataCiteMetadata.Identifier.builder()
060        .withValue(doi.getDoiName())
061        .withIdentifierType("DOI")
062        .build()
063    );
064    LOG.debug("Metadata XML passed validation {}", doi);
065    return toXml(data, true);
066  }
067
068  /**
069   * Produces a validated xml representation of the data cite metadata.
070   * The boolean flag 'validate' it is used to specify if it should be validated.
071   * The generated XML is validated according to the datacite xsd schema.
072   *
073   * @return xml string
074   * @throws InvalidMetadataException if the generated xml failed validation, e.g. missing mandatory elements
075   */
076  public static String toXml(DataCiteMetadata data, boolean validate) throws InvalidMetadataException {
077    // (un)marshaller are not thread safe and need to be created on each authCall
078    try (StringWriter writer = new StringWriter()) {
079      Marshaller m = CONTEXT.createMarshaller();
080      m.setProperty(Marshaller.JAXB_FORMATTED_OUTPUT, Boolean.TRUE);
081      m.setProperty(Marshaller.JAXB_SCHEMA_LOCATION, DATACITE_SCHEMA_LOCATION);
082
083      m.marshal(data, writer);
084      writer.flush();
085
086      String xml = writer.toString();
087      if (validate) {
088        // validate the xml before we return it
089        validateMetadata(xml);
090      }
091      return xml;
092    } catch (JAXBException | IOException e) {
093      throw new InvalidMetadataException(e);
094    }
095  }
096
097  /**
098   * Create metadata from String XML.
099   *
100   * @param xml string metadata
101   * @return DataCiteMetadata
102   * @throws JAXBException in case of unmarshalling problems
103   */
104  public static DataCiteMetadata fromXml(String xml) throws JAXBException {
105    return fromXml(new ByteArrayInputStream(xml.getBytes(Charsets.UTF_8)));
106  }
107
108  /**
109   * Create metadata from byte stream.
110   *
111   * @param in byte stream
112   * @return DataCiteMetadata
113   * @throws JAXBException in case of unmarshalling problems
114   */
115  public static DataCiteMetadata fromXml(InputStream in) throws JAXBException {
116    // (un)marshaller are not thread safe and need to be created on each call
117    Unmarshaller m = CONTEXT.createUnmarshaller();
118    return (DataCiteMetadata) m.unmarshal(in);
119  }
120
121  /**
122   * Validate metadata.
123   *
124   * @param xml DataCite xml metadata to validate (String)
125   * @throws InvalidMetadataException in case of invalid metadata
126   */
127  public static void validateMetadata(String xml) throws InvalidMetadataException {
128    validateMetadata(new StreamSource(new StringReader(xml)));
129  }
130
131  /**
132   * Validate metadata.
133   *
134   * @param xml DataCite xml metadata to validate (byte stream)
135   * @throws InvalidMetadataException in case of invalid metadata
136   */
137  public static void validateMetadata(InputStream xml) throws InvalidMetadataException {
138    validateMetadata(new StreamSource(xml));
139  }
140
141  /**
142   * Validate metadata.
143   *
144   * @param source DataCite xml metadata to validate (abstract source)
145   * @throws InvalidMetadataException in case of invalid metadata
146   */
147  public static void validateMetadata(Source source) throws InvalidMetadataException {
148    try {
149      getValidator().validate(source);
150      LOG.debug("Metadata XML passed validation");
151    } catch (SAXException e) {
152      throw new InvalidMetadataException(e);
153    } catch (IOException e) {
154      // cant happen, we had a string in memory already!
155      throw new IllegalStateException(e);
156    }
157  }
158
159  /**
160   * Lazy create the dataciteSchema and return a new instance of validator on each call.
161   * Validator instances are NOT thread-safe.
162   *
163   * @return validator
164   * @throws SAXException in case of xml parsing exceptions
165   */
166  private synchronized static Validator getValidator() throws SAXException {
167    if (dataciteSchema == null) {
168      String schemaLang = "http://www.w3.org/2001/XMLSchema";
169      // resolve validation driver:
170      SchemaFactory factory = SchemaFactory.newInstance(schemaLang);
171      // create schema by reading it from an URL:
172      dataciteSchema = factory.newSchema(new StreamSource(DATACITE_XSD_LOCATION));
173    }
174
175    return dataciteSchema.newValidator();
176  }
177
178}
