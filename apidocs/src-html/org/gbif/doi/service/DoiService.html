016package org.gbif.doi.service;
017
018import org.gbif.api.model.common.DOI;
019import org.gbif.api.model.common.DoiData;
020import org.gbif.doi.metadata.datacite.DataCiteMetadata;
021
022import java.net.URI;
023
024import javax.annotation.Nonnull;
025
026public interface DoiService {
027
028  /**
029   * Resolves the registered identifier to its status and target URL.
030   *
031   * @param doi the identifier to resolve
032   * @return the status object with the target URL the DOI is backed by or null if DOI does not
033   *     exist at all
034   * @throws DoiException if the operation failed for any reason
035   */
036  @Nonnull
037  DoiData resolve(DOI doi) throws DoiException;
038
039  /**
040   * Check if a DOI is reserved or registered.
041   *
042   * @param doi the identifier to check
043   * @return true if exists or null if not
044   * @throws DoiException if the operation failed for any reason
045   */
046  boolean exists(DOI doi) throws DoiException;
047
048  /**
049   * Get the metadata associated with the doi.
050   *
051   * @param doi the identifier to resolve
052   * @return String XML metadata
053   * @throws DoiException if the operation failed for any reason
054   */
055  String getMetadata(DOI doi) throws DoiException;
056
057  /**
058   * Reserves a new identifier and defines its initial metadata. Reserved ids are not known to
059   * resolvers yet and can still be fully deleted.
060   *
061   * @param doi the identifier to reserve
062   * @param metadata the metadata to be associated with the doi given as a valid DataCite v3.1 XML
063   *     document. The identifier inside the metadata will be overwritten by the doi parameter given
064   * @throws DoiExistsException if the DOI existed already regardless of its status
065   * @throws DoiException if the operation failed for any reason
066   */
067  void reserve(DOI doi, String metadata) throws DoiException;
068
069  /**
070   * Reserves a new identifier and defines its initial metadata. Reserved ids are not known to
071   * resolvers yet and can still be fully deleted.
072   *
073   * @param doi the identifier to reserve
074   * @param metadata the metadata to be associated with the doi. The identifier inside the metadata
075   *     will be overwritten by the doi parameter given
076   * @throws DoiExistsException if the DOI existed already regardless of its status
077   * @throws DoiException if the operation failed for any reason
078   */
079  void reserve(DOI doi, DataCiteMetadata metadata) throws DoiException;
080
081  /**
082   * Registers an identifier that is either brand new, has been reserved or is currently marked as
083   * deleted. It assigns the latest metadata and a URL for resolution. This causes the DOI to be
084   * publicly registered with resolvers and other external services.
085   *
086   * @param doi the identifier to register
087   * @param target the URL the DOI should resolve to
088   * @param metadata the metadata to be associated with the doi given as a valid DataCite v3.1 XML
089   *     document. The identifier inside the metadata will be overwritten by the explicitly given
090   *     DOI
091   * @throws DoiException if the operation failed for any reason
092   * @throws DoiExistsException if the DOI was already registered
093   */
094  void register(DOI doi, URI target, String metadata) throws DoiException;
095
096  /**
097   * Registers an identifier that is either brand new, has been reserved or is currently marked as
098   * deleted. It assigns the latest metadata and a URL for resolution. This causes the DOI to be
099   * publicly registered with resolvers and other external services.
100   *
101   * @param doi the identifier to register
102   * @param target the URL the DOI should resolve to
103   * @param metadata the metadata to be associated with the doi. The identifier inside the metadata
104   *     will be overwritten by the explicitly given DOI
105   * @throws DoiException if the operation failed for any reason
106   * @throws DoiExistsException if the DOI was already registered
107   */
108  void register(DOI doi, URI target, DataCiteMetadata metadata) throws DoiException;
109
110  /**
111   * Tries to delete an identifier. If the DOI has only been reserved it will be fully deleted, if
112   * it was registered before it cannot be deleted as DOIs are permanent identifiers. You can
113   * re-register a deleted DOI again if needed.
114   *
115   * @param doi the identifier to delete
116   * @return true if the reserved DOI was fully deleted, false if it was only marked as deleted
117   * @throws DoiException if the operation failed for any reason
118   */
119  boolean delete(DOI doi) throws DoiException;
120
121  /**
122   * Updates the identifier metadata. This method must be called every time the object or metadata
123   * referenced by the identifier changes (e.g. a dataset gets republished, a dataset is replaced by
124   * a new major version, etc).
125   *
126   * @param doi the identifier of metadata to update
127   * @param metadata the DataCite metadata
128   * @throws DoiException if the operation failed for any reason
129   */
130  void update(DOI doi, String metadata) throws DoiException;
131
132  /**
133   * Updates the identifier metadata. This method must be called every time the object or metadata
134   * referenced by the identifier changes (e.g. a dataset gets republished, a dataset is replaced by
135   * a new major version, etc).
136   *
137   * @param doi the identifier of metadata to update
138   * @param metadata the DataCite metadata
139   * @throws DoiException if the operation failed for any reason
140   */
141  void update(DOI doi, DataCiteMetadata metadata) throws DoiException;
142
143  /**
144   * Updates the registered identifier's target URL.
145   *
146   * @param doi the identifier of metadata to update
147   * @param target the new URL the DOI should resolve to
148   * @throws DoiException if the operation failed for any reason
149   */
150  void update(DOI doi, URI target) throws DoiException;
151}
