001package org.gbif.doi.service;
002
003import org.gbif.api.model.common.DOI;
004import org.gbif.api.model.common.DoiData;
005import org.gbif.doi.metadata.datacite.DataCiteMetadata;
006
007import java.net.URI;
008import javax.annotation.Nullable;
009
010public interface DoiService {
011
012  /**
013   * Resolves the registered identifier to its status and target URL.
014   * @param doi the identifier to resolve
015   * @return the status object with the target URL the DOI is backed by
016   *         or null if DOI does not exist at all
017   *
018   * @throws DoiException if the operation failed for any reason
019   */
020  @Nullable
021  DoiData resolve(DOI doi) throws DoiException;
022
023  /**
024   * Reserves a new identifier and defines its initial metadata.
025   * Reserved ids are not known to resolvers yet and can still be fully deleted.
026   *
027   * @param doi the identifier to reserve
028   * @param metadata the metadata to be associated with the doi given as a valid DataCite v3.1 XML document.
029   *                 The identifier inside the metadata will be overwritten by the doi parameter given
030   *
031   * @throws DoiExistsException if the DOI existed already regardless of its status
032   * @throws DoiException if the operation failed for any reason
033   */
034  void reserve(DOI doi, String metadata) throws DoiException;
035
036  /**
037   * Reserves a new identifier and defines its initial metadata.
038   * Reserved ids are not known to resolvers yet and can still be fully deleted.
039   *
040   * @param doi the identifier to reserve
041   * @param metadata the metadata to be associated with the doi. The identifier inside the metadata will be overwritten
042   *                 by the doi parameter given
043   *
044   * @throws DoiExistsException if the DOI existed already regardless of its status
045   * @throws DoiException if the operation failed for any reason
046   */
047  void reserve(DOI doi, DataCiteMetadata metadata) throws DoiExistsException, DoiException;
048
049  /**
050   * Registers an identifier that is either brand new, has been reserved or is currently marked as deleted.
051   * It assigns the latest metadata and a URL for resolution.
052   * This causes the DOI to be publicly registered with resolvers and other external services.
053   * <br/>
054   * DataCite and EZID restrict target URL domains and the target URL given MUST have a domain
055   * matching your account permissions.
056   * <br/>
057   * You can also call this method to re-register previously deleted, registered DOIs
058   * that are currently marked as inactive (DataCite) or unavailable (EZID). Because the API of these 2 implementations
059   * differ in how to re-register deleted DOIs (DataCite requires new metadata, EZID needs a new terget URL),
060   * we have to take both the metadata and the target URL as params.
061   *
062   * @param doi the identifier to register
063   * @param target the URL the DOI should resolve to
064   * @param metadata the metadata to be associated with the doi given as a valid DataCite v3.1 XML document.
065   *                 The identifier inside the metadata will be overwritten by the explicitly given DOI
066   *
067   * @throws DoiException if the operation failed for any reason
068   * @throws DoiExistsException if the DOI was already registered
069   */
070  void register(DOI doi, URI target, String metadata) throws DoiException;
071
072  /**
073   * Registers an identifier that is either brand new, has been reserved or is currently marked as deleted.
074   * It assigns the latest metadata and a URL for resolution.
075   * This causes the DOI to be publicly registered with resolvers and other external services.
076   * <br/>
077   * DataCite and EZID restrict target URL domains and the target URL given MUST have a domain
078   * matching your account permissions.
079   * <br/>
080   * You can also call this method to re-register previously deleted, registered DOIs
081   * that are currently marked as inactive (DataCite) or unavailable (EZID). Because the API of these 2 implementations
082   * differ in how to re-register deleted DOIs (DataCite requires new metadata, EZID needs a new terget URL),
083   * we have to take both the metadata and the target URL as params.
084   *
085   * @param doi the identifier to register
086   * @param target the URL the DOI should resolve to
087   * @param metadata the metadata to be associated with the doi. The identifier inside the metadata will be overwritten
088   *                 by the explicitly given DOI
089   *
090   * @throws DoiException if the operation failed for any reason
091   * @throws DoiExistsException if the DOI was already registered
092   */
093  void register(DOI doi, URI target, DataCiteMetadata metadata) throws DoiExistsException, DoiException;
094
095  /**
096   * Tries to delete an identifier. If the DOI has only been reserved it will be fully deleted,
097   * if it was registered before it cannot be deleted as DOIs are permanent identifiers.
098   * In DataCite a DOI will be marked as "inactive" though, in EZID as "unavailable".
099   * You can re-register a deleted DOI again if needed.
100   *
101   * @param doi the identifier to delete
102   * @return true if the reserved DOI was fully deleted, false if it was only marked as deleted
103   *
104   * @throws DoiException if the operation failed for any reason
105   */
106  boolean delete(DOI doi) throws DoiException;
107
108  /**
109   * Updates the identifier metadata. This method must be called every time the object or metadata referenced by
110   * the identifier changes (e.g. a dataset gets republished, a dataset is replaced by a new major version, etc).
111   *
112   * @param doi the identifier of metadata to update
113   * @param metadata the datacite metadata given as a valid DataCite v3.1 XML document
114   *
115   * @throws DoiException if the operation failed for any reason
116   */
117  void update(DOI doi, String metadata) throws DoiException;
118
119  /**
120   * Updates the identifier metadata. This method must be called every time the object or metadata referenced by
121   * the identifier changes (e.g. a dataset gets republished, a dataset is replaced by a new major version, etc).
122   *
123   * @param doi the identifier of metadata to update
124   * @param metadata the datacite metadata
125   *
126   * @throws DoiException if the operation failed for any reason
127   */
128  void update(DOI doi, DataCiteMetadata metadata) throws DoiException;
129
130  /**
131   * Updates the registered identifier's target URL.
132   * DataCite and EZID restrict target URL domains and the target URL given MUST have a domain
133   * matching your account permissions.
134   *
135   * @param doi the identifier of metadata to update
136   * @param target the new URL the DOI should resolve to
137   *
138   * @throws DoiException if the operation failed for any reason
139   */
140  void update(DOI doi, URI target) throws DoiException;
141}
