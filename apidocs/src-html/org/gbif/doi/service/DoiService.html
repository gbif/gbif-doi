001package org.gbif.doi.service;
002
003import org.gbif.api.model.common.DOI;
004import org.gbif.doi.metadata.datacite.DataCiteMetadata;
005
006import java.net.URI;
007import javax.annotation.Nullable;
008
009public interface DoiService {
010
011  /**
012   * Resolves the registered identifier to its status and target URL.
013   * @param doi the identifier to resolve
014   * @return the status object with the target URL the DOI is backed by
015   *         or null if DOI does not exist at all
016   *
017   * @throws DoiException if the operation failed for any reason
018   */
019  @Nullable
020  DoiStatus resolve(DOI doi) throws DoiException;
021
022  /**
023   * Reserves a new identifier and defines its initial metadata.
024   * Reserved ids are not known to resolvers yet and can still be fully deleted.
025   *
026   * @param metadata the metadata to be associated with the doi. The identifier inside the metadata will be overwritten
027   *                 by the doi parameter given
028   * @param doi the identifier to reserve
029   *
030   * @throws DoiExistsException if the DOI existed already regardless of its status
031   * @throws DoiException if the operation failed for any reason
032   */
033  void reserve(DOI doi, DataCiteMetadata metadata) throws DoiExistsException, DoiException;
034
035  /**
036   * Registers an identifier that is either brand new, has been reserved or is currently marked as deleted.
037   * It assigns the latest metadata and a URL for resolution.
038   * This causes the DOI to be publicly registered with resolvers and other external services.
039   * <br/>
040   * DataCite and EZID restrict target URL domains and the target URL given MUST have a domain
041   * matching your account permissions.
042   * <br/>
043   * You can also call this method to re-register previously deleted, registered DOIs
044   * that are currently marked as inactive (DataCite) or unavailable (EZID). Because the API of these 2 implementations
045   * differ in how to re-register deleted DOIs (DataCite requires new metadata, EZID needs a new terget URL),
046   * we have to take both the metadata and the target URL as params.
047   *
048   * @param doi the identifier to register
049   * @param target the URL the DOI should resolve to
050   * @param metadata the metadata to be associated with the doi. The identifier inside the metadata will be overwritten
051   *                 by the explicitly given DOI
052   *
053   * @throws DoiException if the operation failed for any reason
054   * @throws DoiExistsException if the DOI was already registered
055   */
056  void register(DOI doi, URI target, DataCiteMetadata metadata) throws DoiExistsException, DoiException;
057
058  /**
059   * Tries to delete an identifier. If the DOI has only been reserved it will be fully deleted,
060   * if it was registered before it cannot be deleted as DOIs are permanent identifiers.
061   * In DataCite a DOI will be marked as "inactive" though, in EZID as "unavailable".
062   * You can re-register a deleted DOI again if needed.
063   *
064   * @param doi the identifier to delete
065   * @return true if the reserved DOI was fully deleted, false if it was only marked as deleted
066   *
067   * @throws DoiException if the operation failed for any reason
068   */
069  boolean delete(DOI doi) throws DoiException;
070
071  /**
072   * Updates the identifier metadata. This method must be called every time the object or metadata referenced by
073   * the identifier changes (e.g. a dataset gets republished, a dataset is replaced by a new major version, etc).
074   *
075   * @param doi the identifier of metadata to update
076   *
077   * @throws DoiException if the operation failed for any reason
078   */
079  void update(DOI doi, DataCiteMetadata metadata) throws DoiException;
080
081  /**
082   * Updates the registered identifier's target URL.
083   * DataCite and EZID restrict target URL domains and the target URL given MUST have a domain
084   * matching your account permissions.
085   *
086   * @param doi the identifier of metadata to update
087   * @param target the new URL the DOI should resolve to
088   *
089   * @throws DoiException if the operation failed for any reason
090   */
091  void update(DOI doi, URI target) throws DoiException;
092}
