001package org.gbif.doi.service;
002
003import java.io.IOException;
004import java.net.URI;
005
006import com.google.common.base.Charsets;
007import org.apache.http.HttpResponse;
008import org.apache.http.auth.AuthScope;
009import org.apache.http.auth.UsernamePasswordCredentials;
010import org.apache.http.client.CredentialsProvider;
011import org.apache.http.client.config.RequestConfig;
012import org.apache.http.client.methods.CloseableHttpResponse;
013import org.apache.http.client.methods.HttpDelete;
014import org.apache.http.client.methods.HttpGet;
015import org.apache.http.client.methods.HttpRequestBase;
016import org.apache.http.client.protocol.HttpClientContext;
017import org.apache.http.entity.ContentType;
018import org.apache.http.impl.client.BasicCredentialsProvider;
019import org.apache.http.impl.client.CloseableHttpClient;
020import org.apache.http.protocol.BasicHttpContext;
021import org.apache.http.protocol.HttpContext;
022import org.apache.http.util.EntityUtils;
023import org.slf4j.Logger;
024import org.slf4j.LoggerFactory;
025
026/**
027 * This class provides common methods to be used by all services.
028 */
029public abstract class BaseService implements DoiService {
030
031  private static final Logger LOG = LoggerFactory.getLogger(BaseService.class);
032  protected static final ContentType PLAIN_TEXT_UTF8 = ContentType.create("text/plain", Charsets.UTF_8);
033
034  private final CloseableHttpClient httpClient;
035  private final CredentialsProvider credsProvider;
036
037  protected BaseService(CloseableHttpClient httpClient, ServiceConfig cfg) {
038    this.httpClient = httpClient;
039    LOG.info("Setup DOI service using account {}", cfg.getUsername());
040    credsProvider = new BasicCredentialsProvider();
041    credsProvider.setCredentials(AuthScope.ANY, new UsernamePasswordCredentials(cfg.getUsername(), cfg.getPassword()));
042  }
043
044  /**
045   * Does an authenticated call and throws a DoiHttpException for non successful responses other than 2xx.
046   * If successful return the entities body as a string.
047   * @throws DoiException
048   */
049  protected String authCall(HttpRequestBase req) throws DoiException {
050
051    RequestConfig.Builder requestConfig = RequestConfig.custom();
052    requestConfig.setConnectTimeout(60 * 1000);
053    requestConfig.setConnectionRequestTimeout(60 * 1000);
054    requestConfig.setSocketTimeout(60 * 1000);
055    req.setConfig(requestConfig.build());
056
057    CloseableHttpResponse resp = null;
058    try {
059      // authentication
060      resp = httpClient.execute(req, buildAuthContext());
061      // Everything but HTTP 2xx is an error
062      if (!isSuccess(resp)) {
063        // read response body and use as error message
064        String msg = EntityUtils.toString(resp.getEntity(), Charsets.UTF_8);
065        LOG.debug("Received HTTP {} {}: {}", resp.getStatusLine().getStatusCode(), resp.getStatusLine().getReasonPhrase(), msg);
066        closeSilently(resp);
067        throw new DoiHttpException(resp.getStatusLine().getStatusCode(), msg);
068      }
069      return EntityUtils.toString(resp.getEntity(), Charsets.UTF_8);
070
071    } catch (DoiException e) {
072      throw e;
073    } catch (Throwable e) {
074      throw new DoiException(e);
075    } finally {
076      closeSilently(resp);
077    }
078  }
079
080  private static void closeSilently(CloseableHttpResponse resp) {
081    if (resp != null) {
082      try {
083        resp.close();
084      } catch (IOException e) {
085        LOG.warn("Could not close http connection", e);
086        // ignore, we're done the request was executed
087      }
088    }
089  }
090
091  /**
092   * Does an authenticated http GET and returns the full body entity as a String
093   */
094  protected String get(URI uri) throws DoiException {
095    return authCall(new HttpGet(uri));
096  }
097
098  /**
099   * Does an authenticated http DELETE and returns the full body entity as a String
100   */
101  protected String delete(URI uri) throws DoiException {
102    return authCall(new HttpDelete(uri));
103  }
104
105  protected static boolean isSuccess(HttpResponse resp) {
106    return Math.round(resp.getStatusLine().getStatusCode() / 100.0) == 2;
107  }
108
109  private HttpContext buildAuthContext() {
110    HttpContext authContext = new BasicHttpContext();
111    authContext.setAttribute(HttpClientContext.CREDS_PROVIDER, credsProvider);
112    return authContext;
113  }
114
115
116}
