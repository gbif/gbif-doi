001package org.gbif.doi.service;
002
003import org.gbif.api.model.common.DOI;
004
005/**
006 * Happening when trying to delete or update a non-existing DOI.
007 */
008public class DoiNotFoundException extends DoiException {
009  private final DOI doi;
010
011  public DoiNotFoundException(Throwable cause, DOI doi) {
012    super(cause);
013    this.doi = doi;
014  }
015
016  public DoiNotFoundException(DOI doi) {
017    this.doi = doi;
018  }
019
020  public DOI getDoi() {
021    return doi;
022  }
023}
