001package org.gbif.doi.service;
002
003import java.net.URI;
004
005import com.google.common.base.Objects;
006
007/**
008 * Simple status object for a DOI with a target URI and a status.
009 */
010public class DoiStatus {
011  private final Status status;
012  private final URI target;
013
014  public DoiStatus(Status status, URI target) {
015    this.status = status;
016    this.target = target;
017  }
018
019  public DoiStatus(String ezidStatus, URI target) {
020    this.status = Status.fromString(ezidStatus);
021    this.target = target;
022  }
023
024  public Status getStatus() {
025    return status;
026  }
027
028  public URI getTarget() {
029    return target;
030  }
031
032  @Override
033  public int hashCode() {
034    return Objects.hashCode(status, target);
035  }
036
037  @Override
038  public boolean equals(Object obj) {
039    if (this == obj) {
040      return true;
041    }
042    if (obj == null || getClass() != obj.getClass()) {
043      return false;
044    }
045    final DoiStatus other = (DoiStatus) obj;
046    return Objects.equal(this.status, other.status) && Objects.equal(this.target, other.target);
047  }
048
049  @Override
050  public String toString() {
051    return "DoiStatus{" +
052           "status=" + status +
053           ", target=" + target +
054           '}';
055  }
056
057  /**
058   * This enumeration represents the status of a DOI identifier.
059   * It represents both EZID and DataCite DOIs.
060   */
061  public enum Status {
062    /**
063     * The identifier is known only to the DOI registration agency.
064     * This status may be used to reserve an identifier name without advertising the identifier's existence
065     * to resolvers and other external services.
066     * A reserved identifier may be fully deleted.
067     */
068    RESERVED("reserved"),
069
070    /**
071     * A public, registered DOI.
072     * It's target URL is known to public resolvers and other external services.
073     * It may be marked as DELETED in the future, but never again as RESERVED.
074     */
075    REGISTERED("public"),
076
077    /**
078     * The identifier once was registered, but the object referenced by the identifier is not available.
079     * This is known as an "inactive" DOI in DataCite and "unavailable" in EZID.
080     * <br/>
081     * In EZID the identifier redirects to an EZID-provided "tombstone" page regardless of its target URL.
082     * In DataCite the original target URL is still available.
083     */
084    DELETED("unavailable");
085
086    private final String ezid;
087
088    private Status(String ezid) {
089      this.ezid = ezid;
090    }
091
092    /**
093     * @return the identifier status value used in EZID.
094     */
095    public String getEzid() {
096      return ezid;
097    }
098
099    /**
100     * @return true if the identifier is registered
101     */
102    public boolean isRegistered() {
103      return this != RESERVED;
104    }
105
106    public static Status fromString(String status) {
107      for (Status s : Status.values()) {
108        if (status.equalsIgnoreCase(s.name()) || status.equalsIgnoreCase(s.getEzid())) {
109          return s;
110        }
111      }
112      return null;
113    }
114  }
115
116}
