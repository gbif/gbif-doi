017/**
018 * ANVL (A Name-Value Language) is a simple record syntax based on email headers.
019 * An ANVL record is a sequence of data elements ending in a blank line.
020 * An element consists of a label, a colon, and an optional value, and a long value may be folded (continued)
021 * onto the next line by inserting a newline and indenting the next line.
022 *
023 * ANVL is used to excahnge metadata with EZID.
024 * This utils class contains methods to (de)serialize them from a java map to a plain string.
025 *
026 * @see <a href="https://wiki.ucop.edu/display/Curation/Anvl">ANVL specs</a>
027 * @see <a href="http://ezid.cdlib.org/doc/apidoc.html#request-response-bodies">EZID API</a>
028 */
029public class AnvlUtils {
030
031  /**
032   * The identifier's target URL. Defaults to the identifier's EZID URL.
033   * That is, the default target URL for identifier foo is the self-referential URL http://ezid.cdlib.org/id/foo.
034   * Note that creating or updating the target URL of a DOI identifier may take up to 30 minutes
035   * to take effect in the Handle System.
036   */
037  public static final String TARGET = "_target";
038  public static final String DATACITE = "datacite";
039  /**
040   * The identifier's status (see Identifier status above).
041   */
042  public static final String STATUS = "_status";
043  /**
044   * The identifier's preferred metadata profile (see Metadata profiles next).
045   */
046  public static final String PROFILE = "_profile";
047  /**
048   * If returned, indicates that the identifier is registered with CrossRef
049   * (or, in the case of a reserved identifier, will be registered),
050   * and also indicates the status of the registration process.
051   * When setting, must be set to "yes" or "no".
052   */
053  public static final String CROSSREF = "_crossref";
054
055  /**
056   * This method serializes a map of metadata name/value pairs into an ANVL String value. If the
057   * map is null, or if it is empty, then it returns a null string.
058   * </br>
059   * EZID metadata dictionaries must be serialized using a subset of A Name-Value Language (ANVL) rules:
060   * -One element name/value pair per line.
061   * -Names separated from values by colons.
062   *
063   * @see <a href="http://ezid.cdlib.org/doc/apidoc.html#request-response-bodies">EZID API</a>
064   * @param metadata the Map of metadata name/value pairs
065   *
066   * @return an ANVL String or null if no metadata name/value pairs provided
067   */
068  public static String write(Map<String, String> metadata) {
069    StringBuffer buffer = new StringBuffer();
070    if (metadata != null) {
071      for (Map.Entry<String, String> entry : metadata.entrySet()) {
072        buffer.append(encode(entry.getKey()) + ": " + encode(entry.getValue()) + "\n");
073      }
074    }
075    return buffer.toString();
076  }
077
078  private static String encode(String s) {
079    return s.replace("%", "%25")
080      .replace("\n", "%0A")
081      .replace("\r", "%0D")
082      .replace(":", "%3A");
083  }
084
085  private static String decode (String s) {
086    StringBuffer b = new StringBuffer();
087    int i;
088    while ((i = s.indexOf("%")) >= 0) {
089      b.append(s.substring(0, i));
090      b.append((char)Integer.parseInt(s.substring(i+1, i+3), 16));
091      s = s.substring(i+3);
092    }
093    b.append(s);
094    return b.toString();
095  }
096
097  private static String[] parseAnvlLine (String line) {
098    String[] kv = line.split(":", 2);
099    kv[0] = decode(kv[0]).trim();
100    kv[1] = decode(kv[1]).trim();
101    return kv;
102  }
103
104  /**
105   * Reads an ANVL input stream into a map.
106   */
107  public static Map<String, String> read(InputStream is) throws IOException {
108    return read(new BufferedReader(new InputStreamReader(is, Charsets.UTF_8)));
109  }
110
111  public static Map<String, String> read(Reader r) throws IOException {
112    return read(new BufferedReader(r));
113  }
114
115  public static Map<String, String> read(String anvl) {
116    try {
117      return read(new BufferedReader(new StringReader(anvl)));
118    } catch (IOException e) {
119      // cant happen so we hide the IOException as a runtime exception
120      // we have a string wrapped reader
121      throw new IllegalStateException(e);
122    }
123  }
124
125  private static Map<String, String> read(BufferedReader br) throws IOException {
126    HashMap<String, String> d = new HashMap<String, String>();
127    String l;
128    while ((l = br.readLine()) != null) {
129      String[] kv = parseAnvlLine(l);
130      d.put(kv[0], kv[1]);
131    }
132    return d;
133  }
134
135  public static class Anvl{
136    private String target;
137
138    public Anvl(String target, String datacite, String status) {
139      this.target = target;
140      this.datacite = datacite;
141      this.status = status;
142    }
143
144    private String datacite;
145    private String status;
146  }
147
148  public static class Builder {
149
150    private Map<String, String> map = new HashMap<String, String>();
151
152    public Builder target(URI target) {
153      if (target == null) {
154        this.map.put(AnvlUtils.TARGET, "");
155      } else {
156        this.map.put(AnvlUtils.TARGET, target.toString());
157      }
158      return this;
159    }
160
161    public Builder datacite(String datacite) {
162      this.map.put(AnvlUtils.DATACITE, datacite);
163      return this;
164    }
165
166    public Builder status(DoiStatus.Status status) {
167      this.map.put(AnvlUtils.STATUS, status.getEzid());
168      return this;
169    }
170
171    public Builder entry(String key, String value) {
172      this.map.put(key, value);
173      return this;
174    }
175
176    public Map<String, String> build() {
177      return map;
178    }
179  }
180
181  public static Builder builder() {
182    return new Builder();
183  }
