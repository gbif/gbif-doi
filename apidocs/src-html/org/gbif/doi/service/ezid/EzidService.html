013import java.net.URI;
014import java.util.Map;
015
016import com.google.common.annotations.VisibleForTesting;
017import com.google.common.base.Preconditions;
018import com.google.common.collect.Maps;
019import org.apache.http.HttpEntity;
020import org.apache.http.client.methods.HttpEntityEnclosingRequestBase;
021import org.apache.http.client.methods.HttpPost;
022import org.apache.http.client.methods.HttpPut;
023import org.apache.http.entity.StringEntity;
024import org.apache.http.impl.client.CloseableHttpClient;
025import org.slf4j.Logger;

028/**
029 * This EZID service implementation.
030 * TODO: offer option to also register DOIs with CrossRef which is supported by the EZID API
031 * http://ezid.cdlib.org/doc/apidoc.html
032 */
033public class EzidService extends BaseService {
034
035  private static final Logger LOG = LoggerFactory.getLogger(EzidService.class);
036  private static final URI DEFAULT_API = URI.create("https://ezid.cdlib.org/");
037  private static final String NON_TARGET = "http://ezid.cdlib.org/id";
038  private final URI apiWs;
039  private final URI idWs;
040
041
042  public EzidService(CloseableHttpClient httpClient, ServiceConfig cfg) {
043    super(httpClient, cfg);
044    apiWs = cfg.getApi() == null ? DEFAULT_API : cfg.getApi();
045    idWs = apiWs.resolve("id/");
046  }
047
048  /**
049   * This creates the EZID URI, and ensures that it is formatted correctly, e.g.
050   * E.g. http://ezid.cdlib.org/id/doi:10.nnnn/suffix
051   * Notice the DOI must start with the DOI access schema ("doi:") and uses an unescaped colon!
052   *
053   * @param doi identifier
054   * @return EZID ID URI
055   */
056  @VisibleForTesting
057  protected URI idUri(DOI doi) throws DoiException {
058    try {
059      return URI.create(idWs.toString() + doi);
060    } catch (IllegalArgumentException e) {
061      throw new DoiException("Unable to convert DOI to EZID URI", e);
062    }
063  }
064
065  @Override
066  public DoiStatus resolve(DOI doi) throws DoiException {
067    Preconditions.checkNotNull(doi);
068    Map<String, String> meta = getANVL(doi);
069    if (meta.containsKey(AnvlUtils.STATUS)) {
070      URI target = null;
071      String url = meta.get(AnvlUtils.TARGET);
072      // avoid the default EZID target url
073      if (url != null && !url.toLowerCase().startsWith(NON_TARGET)) {
074        try {
075          target = URI.create(meta.get(AnvlUtils.TARGET));
076        } catch (IllegalArgumentException e) {
077          LOG.debug("Invalid DOI URL", e.getMessage());
078        }
079      }
080      return new DoiStatus(meta.get(AnvlUtils.STATUS), target);
081    }
082    return null;
083  }
084
085  private Map<String, String> getANVL(DOI doi) throws DoiException {
086    Preconditions.checkNotNull(doi);
087    try {
088      return AnvlUtils.read(get(idUri(doi)));
089
090    } catch (DoiHttpException e) {
091      if (Math.round( e.getStatus() / 100.0) == 4) {
092        LOG.debug("Non existing DOI", e.getMessage());
093        return Maps.newHashMap();
094      }
095      throw e;
096    }
097  }
098
099  @Override
100  public void reserve(DOI doi, DataCiteMetadata metadata) throws DoiException {
101    Preconditions.checkNotNull(doi);
102    Preconditions.checkNotNull(metadata);
103    Map<String, String> data = AnvlUtils.builder()
104      .datacite(DataCiteValidator.toXml(doi, metadata))
105      .status(DoiStatus.Status.RESERVED)
106      .build();
107    postOrPut(doi, data, new HttpPut(idUri(doi)));
108    LOG.info("Reserved {}", doi);
109  }
110
111  @Override
112  public void register(DOI doi, URI target, DataCiteMetadata metadata) throws DoiException {
113    Preconditions.checkNotNull(doi);
114    Preconditions.checkNotNull(target);
115
116    DoiStatus status = resolve(doi);
117    if (status == null) {
118      reserve(doi, metadata);
119    } else if (DoiStatus.Status.REGISTERED == status.getStatus()) {
120      throw new DoiExistsException(doi);
121    }
122    Map<String, String> data = AnvlUtils.builder()
123      .target(target)
124      .datacite(DataCiteValidator.toXml(doi, metadata))
125      .status(DoiStatus.Status.REGISTERED)
126      .build();
127    postOrPut(doi, data, new HttpPost(idUri(doi)));
128    LOG.info("Registered {}", doi);
129  }
130
131  @Override
132  public boolean delete(DOI doi) throws DoiException {
133    DoiStatus status = resolve(doi);
134    if (status != null) {
135      if (DoiStatus.Status.RESERVED == status.getStatus()) {
136        delete(idUri(doi));
137        LOG.info("Deleted {}", doi);
138        return true;
139
140      } else {
141        // this is a registered DOI, we can only change the status to unavailable
142        Map<String, String> data = AnvlUtils.builder()
143          .target(null)
144          .status(DoiStatus.Status.DELETED)
145          .build();
146        postOrPut(doi, data, new HttpPost(idUri(doi)));
147        LOG.info("Made {} unavailable", doi);
148        return false;
149      }
150    }
151    throw new DoiException("DOI " + doi + " not existing");
152  }
153
154  @Override
155  public void update(DOI doi, DataCiteMetadata metadata) throws DoiException {
156    Preconditions.checkNotNull(doi);
157    Preconditions.checkNotNull(metadata);
158    Map<String, String> data = AnvlUtils.builder()
159      .datacite(DataCiteValidator.toXml(doi, metadata))
160      .build();
161    postOrPut(doi, data, new HttpPost(idUri(doi)));
162    LOG.info("Updated metadata for {}", doi);
163  }
164
165  @Override
166  public void update(DOI doi, URI target) throws DoiException {
167    Preconditions.checkNotNull(doi);
168    Preconditions.checkNotNull(target);
169    Map<String, String> data = AnvlUtils.builder()
170      .target(target)
171      .build();
172    postOrPut(doi, data, new HttpPost(idUri(doi)));
173    LOG.info("Updated metadata for {}", doi);
174  }
175
176  /**
177   * Issues a post or put with the given EZID data map for a DOI.
178   */
179  protected void postOrPut(DOI doi, Map<String, String> anvl, HttpEntityEnclosingRequestBase req) throws DoiException {
180    // body
181    if (anvl != null && !anvl.isEmpty()) {
182      HttpEntity entity = new StringEntity(AnvlUtils.write(anvl), PLAIN_TEXT_UTF8);
183      req.setEntity(entity);
184      authCall(req);
185    }
186  }
