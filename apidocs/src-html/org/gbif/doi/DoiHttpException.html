001package org.gbif.doi;
002
003public class DoiHttpException extends DoiException {
004  private final int status;
005
006  public DoiHttpException(int status) {
007    super("HTTP "+ status);
008    this.status = status;
009  }
010
011  public DoiHttpException(Throwable cause, int status) {
012    super("HTTP "+ status, cause);
013    this.status = status;
014  }
015
016  public int getStatus() {
017    return status;
018  }
019}
