016package org.gbif.doi.util;
017
018import org.gbif.doi.metadata.datacite.DataCiteMetadata;
019import org.gbif.doi.service.datacite.DataCiteValidator;
020
021import java.lang.reflect.Field;
022import java.util.LinkedHashSet;
023import java.util.Objects;
024import java.util.Set;
025
026import javax.xml.bind.JAXBException;
027
028import org.slf4j.Logger;
029import org.slf4j.LoggerFactory;
030
031/** Utils related {@link DataCiteMetadata}. */
032public final class MetadataUtils {
033
034  private static final Logger LOG = LoggerFactory.getLogger(MetadataUtils.class);
035
036  private MetadataUtils() {}
037
038  /**
039   * Compare two DataCite metadata objects. Unlike {@link DataCiteMetadata#equals(Object)} the
040   * method is case-insensitive for {@link DataCiteMetadata.Identifier#getValue()}. This is needed
041   * because the DataCite keeps DOIs in uppercase, and GBIF does in lowercase.
042   */
043  public static boolean metadataEquals(DataCiteMetadata metadata1, DataCiteMetadata metadata2) {
044    if (metadata1 == null && metadata2 == null) {
045      return true;
046    }
047
048    if (metadata1 == null || metadata2 == null) {
049      return false;
050    }
051
052    return metadata1
053            .getIdentifier()
054            .getIdentifierType()
055            .equals(metadata2.getIdentifier().getIdentifierType())
056        && metadata1
057            .getIdentifier()
058            .getValue()
059            .equalsIgnoreCase(metadata2.getIdentifier().getValue())
060        && Objects.equals(metadata1.getCreators(), metadata2.getCreators())
061        && Objects.equals(metadata1.getTitles(), metadata2.getTitles())
062        && Objects.equals(metadata1.getPublisher(), metadata2.getPublisher())
063        && Objects.equals(metadata1.getPublicationYear(), metadata2.getPublicationYear())
064        && Objects.equals(metadata1.getResourceType(), metadata2.getResourceType())
065        && Objects.equals(metadata1.getSubjects(), metadata2.getSubjects())
066        && Objects.equals(metadata1.getContributors(), metadata2.getContributors())
067        && Objects.equals(metadata1.getDates(), metadata2.getDates())
068        && Objects.equals(metadata1.getLanguage(), metadata2.getLanguage())
069        && Objects.equals(metadata1.getAlternateIdentifiers(), metadata2.getAlternateIdentifiers())
070        && Objects.equals(metadata1.getRelatedIdentifiers(), metadata2.getRelatedIdentifiers())
071        && Objects.equals(metadata1.getSizes(), metadata2.getSizes())
072        && Objects.equals(metadata1.getFormats(), metadata2.getFormats())
073        && Objects.equals(metadata1.getVersion(), metadata2.getVersion())
074        && Objects.equals(metadata1.getRightsList(), metadata2.getRightsList())
075        && Objects.equals(metadata1.getDescriptions(), metadata2.getDescriptions())
076        && Objects.equals(metadata1.getGeoLocations(), metadata2.getGeoLocations())
077        && Objects.equals(metadata1.getFundingReferences(), metadata2.getFundingReferences());
078  }
079
080  /**
081   * Compare two metadata objects. First convert them to {@link DataCiteMetadata} and apply {@link
082   * MetadataUtils#metadataEquals(DataCiteMetadata, DataCiteMetadata)} afterwards.
083   */
084  public static boolean metadataEquals(String xmlMetadata1, String xmlMetadata2) {
085    try {
086      DataCiteMetadata dataCiteMetadata1 =
087          xmlMetadata1 != null ? DataCiteValidator.fromXml(xmlMetadata1) : null;
088      DataCiteMetadata dataCiteMetadata2 =
089          xmlMetadata2 != null ? DataCiteValidator.fromXml(xmlMetadata2) : null;
090
091      return metadataEquals(dataCiteMetadata1, dataCiteMetadata2);
092    } catch (JAXBException e) {
093      LOG.error("Invalid metadata", e);
094    }
095
096    return false;
097  }
098
099  /** Compare two metadata objects and get {@link Difference} between them. */
100  public static Difference metadataDifference(String xmlMetadata1, String xmlMetadata2) {
101    Set<Difference.DifferenceItem> result = new LinkedHashSet<>();
102
103    try {
104      DataCiteMetadata dataCiteMetadata1 =
105          xmlMetadata1 != null ? DataCiteValidator.fromXml(xmlMetadata1) : null;
106      DataCiteMetadata dataCiteMetadata2 =
107          xmlMetadata2 != null ? DataCiteValidator.fromXml(xmlMetadata2) : null;
108
109      if (dataCiteMetadata1 == null || dataCiteMetadata2 == null) {
110        LOG.error("Can't get difference: null argument(s)");
111      } else {
112        Field[] declaredFields = DataCiteMetadata.class.getDeclaredFields();
113        for (Field field : declaredFields) {
114          field.setAccessible(true);
115          Object value1 = field.get(dataCiteMetadata1);
116          Object value2 = field.get(dataCiteMetadata2);
117
118          if (!Objects.equals(value1, value2)) {
119            result.add(new Difference.DifferenceItem(field.getName(), value1, value2));
120          }
121        }
122      }
123    } catch (JAXBException e) {
124      LOG.error("Invalid metadata", e);
125    } catch (IllegalAccessException e) {
126      LOG.error("Access exception", e);
127    }
128
129    return new Difference(result);
130  }
131}
