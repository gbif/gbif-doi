014package org.gbif.doi.util;
015
016import org.gbif.api.annotation.Generated;
017
018import java.util.Collections;
019import java.util.Objects;
020import java.util.Set;
021import java.util.stream.Collectors;
022
023/** Difference between two objects. This object is intended to use for metadata comparing. */
024public class Difference {
025
026  private static final String DIFFERENCE_DELIMITER = "--------------------";
027
028  private final Status comparisonStatus;
029  private final Set<DifferenceItem> items;
030
031  public Difference(Status status, Set<DifferenceItem> items) {
032    this.comparisonStatus = status;
033    this.items = items;
034  }
035
036  public Status getComparisonStatus() {
037    return comparisonStatus;
038  }
039
040  public Set<DifferenceItem> getItems() {
041    return Collections.unmodifiableSet(items);
042  }
043
044  @Override
045  @Generated
046  public String toString() {
047    if (items.isEmpty()) {
048      return "Status: " + comparisonStatus + "\n";
049    } else {
050      return "Status: "
051          + comparisonStatus
052          + "\n"
053          + items.stream()
054              .map(DifferenceItem::toString)
055              .collect(
056                  Collectors.joining(
057                      DIFFERENCE_DELIMITER, DIFFERENCE_DELIMITER, DIFFERENCE_DELIMITER));
058    }
059  }
060
061  public static class DifferenceItem {
062
063    private final String fieldName;
064    private final Object value1;
065    private final Object value2;
066
067    public DifferenceItem(String fieldName, Object value1, Object value2) {
068      this.fieldName = fieldName;
069      this.value1 = value1;
070      this.value2 = value2;
071    }
072
073    public String getFieldName() {
074      return fieldName;
075    }
076
077    public Object getValue1() {
078      return value1;
079    }
080
081    public Object getValue2() {
082      return value2;
083    }
084
085    @Override
086    @Generated
087    public boolean equals(Object o) {
088      if (this == o) {
089        return true;
090      }
091      if (o == null || getClass() != o.getClass()) {
092        return false;
093      }
094      DifferenceItem that = (DifferenceItem) o;
095      return Objects.equals(fieldName, that.fieldName)
096          && Objects.equals(value1, that.value1)
097          && Objects.equals(value2, that.value2);
098    }
099
100    @Override
101    @Generated
102    public int hashCode() {
103      return Objects.hash(fieldName, value1, value2);
104    }
105
106    @Override
107    @Generated
108    public String toString() {
109      return '\n' + fieldName + '\n' + value1 + '\n' + value2 + '\n';
110    }
111  }
112
113  public enum Status {
114    INCOMPATIBLE_OBJECTS("Objects are incompatible"),
115    NULL_OBJECTS("One or both objects are nulls"),
116    COMPATIBLE_OBJECTS("Compatible objects");
117
118    Status(String value) {
119      this.value = value;
120    }
121
122    private final String value;
123
124    public String getValue() {
125      return value;
126    }
127  }
128}
