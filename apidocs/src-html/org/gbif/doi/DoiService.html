001package org.gbif.doi;
002
003import org.gbif.api.model.common.DOI;
004import org.gbif.doi.metadata.datacite.DataCiteMetadata;
005
006import java.net.URI;
007import javax.annotation.Nullable;
008
009public interface DoiService {
010
011  /**
012   * Resolves the registered identifier to its URL.
013   * @param doi the identifier to get metadata for
014   * @return the URL the DOI is backed by or null if DOI does not exist or is only reserved
015   *
016   * @throws DoiException if the operation failed for any reason
017   */
018  @Nullable
019  URI get(DOI doi) throws DoiException;
020
021  /**
022   * Reserves a new identifier. The identifier status is set to RESERVED.
023   * </br>
024   * With DataCite, this is done by sending a POST request with the XML metadata (as the request body) to
025   * https://mds.xsd.datacite.org/metadata. The DOI is specified inside the metadata.
026   * </br>
027   * With EZID, this is done by sending a PUT request to http://ezid.cdlib.org/id/{id} where {id} is a unique id
028   * with shoulder/namespace prefix, and by setting the "_status" metadata element to "reserved".
029   *
030   * @param metadata the metadata to be associated with the doi. The identifier inside the metadata will be overwritten
031   *                 by the doi parameter given
032   * @param doi the identifier to reserve
033   *
034   * @throws DoiException if the operation failed for any reason
035   */
036  void reserve(DOI doi, DataCiteMetadata metadata) throws DoiException;
037
038  /**
039   * Mints a new random identifier with the given DOI prefix and reserves it.
040   *
041   * @param metadata the metadata to be associated with the doi. The identifier inside the metadata will be overwritten
042   *                 by the generated DOI
043   * @param prefix the identifier prefix to use to generate the random DOI
044   * @param length the length of the randomly generated suffix string. minimum 3, 5-10 is recommended.
045   *
046   * @throws DoiException if the operation failed for any reason
047   */
048  DOI reserveRandom(String prefix, int length, DataCiteMetadata metadata) throws DoiException;
049
050  /**
051   * Registers an identifier that has been reserved and assigns it a URL for resolution.
052   * This causes the DOI to be publicly registered with resolvers and other external services.
053   * DataCite and EZID restrict target URL domains and the target URL given MUST have a domain
054   * matching your account permissions.
055   *
056   * @param doi the identifier to register
057   * @param target the URL the DOI should resolve to
058   *
059   * @throws DoiException if the operation failed for any reason
060   */
061  void register(DOI doi, URI target) throws DoiException;
062
063  /**
064   * Tries to delete an identifier. If the DOI has only been reserved it will be fully deleted,
065   * if it was registered before it's status will become unavailable.
066   * You can register an unavailable DOI again if needed.
067   * </br>
068   * With DataCite, this is done by sending a DELETE request to https://mds.xsd.datacite.org/metadata, which inactivates
069   * the DOI.
070   * </br>
071   * With EZID, this is done by sending a DELETE request to http://ezid.cdlib.org/id/{id}
072   * or by setting the "_status" metadata element to "unavailable" for registered DOIs.
073   *
074   * @param doi the identifier to delete
075   * @return true if the DOI was fully deleted, false if it became unavailable
076   *
077   * @throws DoiException if the operation failed for any reason
078   */
079  boolean delete(DOI doi) throws DoiException;
080
081  /**
082   * Updates the identifier metadata. This method must be called every time the object or metadata referenced by
083   * the identifier changes (e.g. a gets republished, a data set is replaced by a new major version, etc).
084   * </br>
085   * With DataCite, this is done by sending a POST request with XML metadata (as the request body) to
086   * https://mds.xsd.datacite.org/metadata.
087   * </br>
088   * With EZID, this is done by sending a POST request with metadata (as request body) to
089   * http://ezid.cdlib.org/id/{id}.
090   *
091   * @param doi the identifier of metadata to update
092   *
093   * @throws DoiException if the operation failed for any reason
094   */
095  void update(DOI doi, DataCiteMetadata metadata) throws DoiException;
096
097  /**
098   * Updates the registered identifiers target URL.
099   * DataCite and EZID restrict target URL domains and the target URL given MUST have a domain
100   * matching your account permissions.
101   *
102   * @param doi the identifier of metadata to update
103   * @param target the new URL the DOI should resolve to
104   *
105   * @throws DoiException if the operation failed for any reason
106   */
107  void update(DOI doi, URI target) throws DoiException;
108}
