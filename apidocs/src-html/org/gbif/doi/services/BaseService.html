001package org.gbif.doi.services;
002
003import org.gbif.api.model.common.DOI;
004import org.gbif.doi.DoiException;
005import org.gbif.doi.metadata.datacite.DataCiteMetadata;
006
007import java.io.StringWriter;
008import java.net.URI;
009import javax.xml.bind.JAXBContext;
010import javax.xml.bind.JAXBException;
011import javax.xml.bind.Marshaller;
012
013import com.google.common.annotations.VisibleForTesting;
014import com.google.common.base.Charsets;
015import com.google.common.net.MediaType;
016import org.apache.commons.lang3.RandomStringUtils;
017import org.apache.http.HttpEntity;
018import org.apache.http.HttpResponse;
019import org.apache.http.auth.AuthScope;
020import org.apache.http.auth.UsernamePasswordCredentials;
021import org.apache.http.client.CredentialsProvider;
022import org.apache.http.client.HttpClient;
023import org.apache.http.client.methods.HttpDelete;
024import org.apache.http.client.methods.HttpEntityEnclosingRequestBase;
025import org.apache.http.client.methods.HttpPost;
026import org.apache.http.client.methods.HttpPut;
027import org.apache.http.client.protocol.ClientContext;
028import org.apache.http.entity.ByteArrayEntity;
029import org.apache.http.impl.client.BasicCredentialsProvider;
030import org.apache.http.protocol.BasicHttpContext;
031import org.apache.http.protocol.HTTP;
032import org.apache.http.protocol.HttpContext;
033import org.slf4j.Logger;
034import org.slf4j.LoggerFactory;
035
036/**
037 * This class provides common methods to be used by all services.
038 */
039public abstract class BaseService {
040
041  private static final Logger LOG = LoggerFactory.getLogger(BaseService.class);
042  private final HttpClient httpClient;
043  protected final UsernamePasswordCredentials credentials;
044  private final JAXBContext context;
045  protected BaseService(HttpClient httpClient, String username, String password) {
046    this.httpClient = httpClient;
047    this.credentials = new UsernamePasswordCredentials(username, password);
048    try {
049      context = JAXBContext.newInstance(DataCiteMetadata.class);
050    } catch (JAXBException e) {
051      throw new IllegalStateException("Fail to setup JAXB", e);
052    }
053  }
054
055  /**
056   * Executes an HTTP POST request.
057   *
058   * @param uri  to issue request against
059   * @param data request body content to be serialized as XML with jackson
060   *
061   * @throws org.gbif.doi.DoiException in case anything goes wrong during the request.
062   */
063  protected void post(URI uri, DataCiteMetadata data) throws DoiException {
064    LOG.info("POST: {}", uri);
065    call(uri, data, new HttpPost(uri));
066  }
067
068  /**
069   * Executes an HTTP PUT request.
070   *
071   * @param uri  to issue request against
072   * @param data request body content to be serialized as XML with jackson
073   *
074   * @throws org.gbif.doi.DoiException in case anything goes wrong during the request.
075   */
076  protected void put(URI uri, DataCiteMetadata data) throws DoiException {
077    LOG.info("PUT: {}", uri);
078    call(uri, data, new HttpPut(uri));
079  }
080
081  protected void delete(URI uri) throws DoiException {
082    LOG.info("DELETE: {}", uri);
083    try {
084      HttpContext authContext = buildAuthContext(uri, credentials);
085      HttpDelete del = new HttpDelete(uri);
086      HttpResponse response = httpClient.execute(del, authContext);
087    } catch (Throwable e) {
088      throw new DoiException(e);
089    }
090  }
091
092  @VisibleForTesting
093  protected String toXml(DataCiteMetadata data) throws JAXBException {
094    // (un)marshaller are not thread safe and need to be created on each call
095    Marshaller m = context.createMarshaller();
096    m.setProperty(Marshaller.JAXB_FORMATTED_OUTPUT, Boolean.TRUE);
097
098    StringWriter writer = new StringWriter();
099    m.marshal(data, writer);
100    writer.flush();
101    return writer.toString();
102  }
103
104  private void call(URI uri, DataCiteMetadata data, HttpEntityEnclosingRequestBase req) throws DoiException {
105    try {
106      req.addHeader(HTTP.CONTENT_TYPE, MediaType.APPLICATION_XML_UTF_8.toString());
107      // body
108      if (data != null) {
109        HttpEntity entity = new ByteArrayEntity(toXml(data).getBytes(Charsets.UTF_8));
110        req.setEntity(entity);
111      }
112
113      // authentication
114      HttpContext authContext = buildAuthContext(uri, credentials);
115      HttpResponse response = httpClient.execute(req, authContext);
116
117      // Everything but HTTP status 201 is an error
118      if (response.getStatusLine().getStatusCode() != 201) {
119        LOG.debug("Received HTTP code[{}] cause[{}] for request: {}", response.getStatusLine().getStatusCode(),
120          response.getStatusLine().getReasonPhrase(), uri);
121        String cause = String.format("Received HTTP code[%d], phrase[%s]", response.getStatusLine().getStatusCode(),
122          response.getStatusLine().getReasonPhrase());
123        throw new DoiException(cause);
124      }
125
126    } catch (DoiException e) {
127      throw e;
128    } catch (Throwable e) {
129      throw new DoiException(e);
130    }
131  }
132
133  private HttpContext buildAuthContext(URI uri, UsernamePasswordCredentials credentials) {
134    HttpContext authContext = new BasicHttpContext();
135    if (credentials != null) {
136      AuthScope scope = new AuthScope(uri.getHost(), AuthScope.ANY_PORT, AuthScope.ANY_REALM);
137
138      CredentialsProvider credsProvider = new BasicCredentialsProvider();
139      credsProvider.setCredentials(scope, credentials);
140
141      authContext.setAttribute(ClientContext.CREDS_PROVIDER, credsProvider);
142    }
143
144    return authContext;
145  }
146
147  protected void updateMetadataIdentifier(DOI doi, DataCiteMetadata metadata) {
148    metadata.setIdentifier(
149      DataCiteMetadata.Identifier.builder().withValue(doi.toString()).withIdentifierType("DOI").build()
150    );
151  }
152
153  /**
154   * @return a random DOI with the given prefix. It is not guaranteed to be unique and might exist already
155   */
156  protected DOI random(String prefix) {
157    String suffix = RandomStringUtils.randomAlphanumeric(8);
158    DOI doi = new DOI(prefix, suffix);
159    return doi;
160  }
161}
