043/**
044 * This class provides common methods to be used by all services.
045 */
046public abstract class BaseService {
047
048  private static final Logger LOG = LoggerFactory.getLogger(BaseService.class);
049  private static final ContentType APPLICATION_XML_UTF8 = ContentType.create("application/xml", Charsets.UTF_8);
050  private static final String DATACITE_SCHEMA_LOCATION = "http://datacite.org/schema/kernel-3 http://schema.datacite.org/meta/kernel-3/metadata.xsd";
051
052  private final CloseableHttpClient httpClient;
053  private final CredentialsProvider credsProvider;
054  private final JAXBContext context;
055
056  protected BaseService(CloseableHttpClient httpClient, ServiceConfig cfg) {
057    this.httpClient = httpClient;
058    LOG.info("Setup DOI service using account {}", cfg.getUsername());
059    credsProvider = new BasicCredentialsProvider();
060    credsProvider.setCredentials(AuthScope.ANY, new UsernamePasswordCredentials(cfg.getUsername(), cfg.getPassword()));
061    try {
062      context = JAXBContext.newInstance(DataCiteMetadata.class);
063    } catch (JAXBException e) {
064      throw new IllegalStateException("Fail to setup JAXB", e);
065    }
066  }
067
068  /**
069   * Executes an HTTP POST request.
070   *
071   * @param uri  to issue request against
072   * @param data request body content to be serialized as XML with jackson
073   *
074   * @throws org.gbif.doi.DoiException in case anything goes wrong during the request.
075   */
076  protected void post(DOI doi, URI uri, DataCiteMetadata data) throws DoiException {
077    LOG.debug("POST: {}", uri);
078    postOrPut(doi, data, new HttpPost(uri));
079  }
080
081  /**
082   * Executes an HTTP PUT request.
083   *
084   * @param uri  to issue request against
085   * @param data request body content to be serialized as XML with jackson
086   *
087   * @throws org.gbif.doi.DoiException in case anything goes wrong during the request.
088   */
089  protected void put(DOI doi, URI uri, DataCiteMetadata data) throws DoiException {
090    LOG.debug("PUT: {}", uri);
091    postOrPut(doi, data, new HttpPut(uri));
092  }
093
094  protected void delete(URI uri) throws DoiException {
095    LOG.debug("DELETE: {}", uri);
096    try {
097      HttpDelete del = new HttpDelete(uri);
098      CloseableHttpResponse resp = httpClient.execute(del, buildAuthContext());
099      resp.close();
100    } catch (Throwable e) {
101      throw new DoiException(e);
102    }
103  }
104
105  @VisibleForTesting
106  protected String toXml(DataCiteMetadata data) throws JAXBException {
107    // (un)marshaller are not thread safe and need to be created on each authCall
108    Marshaller m = context.createMarshaller();
109    m.setProperty(Marshaller.JAXB_FORMATTED_OUTPUT, Boolean.TRUE);
110    m.setProperty(Marshaller.JAXB_SCHEMA_LOCATION, DATACITE_SCHEMA_LOCATION);
111
112    StringWriter writer = new StringWriter();
113    m.marshal(data, writer);
114    writer.flush();
115    return writer.toString();
116  }
117
118  protected void postOrPut(DOI doi, DataCiteMetadata data, HttpEntityEnclosingRequestBase req) throws DoiException {
119    // body
120    if (data != null) {
121      try {
122        String xml = toXml(data);
123        // validate the xml before we send it to datacite
124        try {
125          DataciteValidator.validateMetadata(xml);
126          LOG.debug("Metadata XML passed validation for ID: {}", doi);
127        } catch (SAXException e) {
128          throw new InvalidMetadataException(doi, e);
129        }
130        HttpEntity entity = new StringEntity(xml, APPLICATION_XML_UTF8);
131        req.setEntity(entity);
132        authCall(req);
133
134      } catch (Exception e) {
135        throw new DoiException(e);
136      }
137    }
138  }
139
140  protected void authCall(HttpUriRequest req) throws DoiException {
141    CloseableHttpResponse resp = authCallWithResponse(req);
142    closeSilently(resp);
143  }
144
145  private static void closeSilently(CloseableHttpResponse resp) {
146    if (resp != null) {
147      try {
148        resp.close();
149      } catch (IOException e) {
150        LOG.warn("Could not close http connection", e);
151        // ignore, we're done the request was executed
152      }
153    }
154  }
155
156  /**
157   * Does an authenticated call and throws a DoiHttpException for nun successful responses other than 2xx.
158   * This returns an open http connection, make sure to close the response!!!
159   * @throws DoiException
160   */
161  private CloseableHttpResponse authCallWithResponse(HttpUriRequest req) throws DoiException {
162    CloseableHttpResponse resp = null;
163    try {
164      // authentication
165      resp = httpClient.execute(req, buildAuthContext());
166      // Everything but HTTP 2xx is an error
167      if (Math.round( resp.getStatusLine().getStatusCode() / 100.0) != 2) {
168        LOG.debug("Received HTTP {}: {}", resp.getStatusLine().getStatusCode(), resp.getStatusLine().getReasonPhrase());
169        closeSilently(resp);
170        throw new DoiHttpException(resp.getStatusLine().getStatusCode());
171      }
172      return resp;
173
174    } catch (DoiException e) {
175      throw e;
176    } catch (Throwable e) {
177      throw new DoiException(e);
178    }
179  }
180
181  /**
182   * Does an http GET and returns the full body entity as a String
183   */
184  protected String get(URI uri) throws DoiException {
185    CloseableHttpResponse resp = authCallWithResponse(new HttpGet(uri));
186    try {
187      String content = EntityUtils.toString(resp.getEntity(), Charsets.UTF_8);
188      return content;
189    } catch (IOException e) {
190      throw new DoiException(e);
191    } finally {
192      closeSilently(resp);
193    }
194  }
195
196  private HttpContext buildAuthContext() {
197    HttpContext authContext = new BasicHttpContext();
198    authContext.setAttribute(ClientContext.CREDS_PROVIDER, credsProvider);
199    return authContext;
200  }
201
202  protected static void updateMetadataIdentifier(DOI doi, DataCiteMetadata metadata) {
203    metadata.setIdentifier(
204      DataCiteMetadata.Identifier.builder().withValue(doi.getDoiName()).withIdentifierType("DOI").build());
205  }
206
207  /**
208   * @return a random DOI with the given prefix. It is not guaranteed to be unique and might exist already
209   */
210  protected static DOI random(String prefix, int length) {
211    String suffix = RandomStringUtils.randomAlphanumeric(length);
212    DOI doi = new DOI(prefix, suffix);
213    return doi;
214  }
215
216  protected static URI doiUri(URI base, DOI doi) {
217    return URI.create(base.toString()+"/"+doi.getDoiName());
218  }
