049  @Override
050  public DOI reserveRandom(String prefix, DataCiteMetadata metadata) throws DoiException {
051    // a random DOI might exist already so we need to try to reserve one until we find a good one.
052    // to avoid endless loop do a max of 100 tries though, we should never ever reach that
053    for (int x=1; x<100; x++) {
054      DOI doi = random(prefix);
055      try {
056        reserve(doi, metadata);
057        return doi;
058      } catch (DoiExistsException e) {
059        // this doi did exist already, mint another one
060      }
061    }
062    LOG.error("Failed to generate a new, random DOI for prefix {}. We have tried 100 random suffices.", prefix);
063    throw new DoiException("Failed to generate a new, random DOI for prefix " + prefix);
064  }
