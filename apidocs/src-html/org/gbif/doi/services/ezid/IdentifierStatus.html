001package org.gbif.doi.services.ezid;
002
003/**
004 * This enumeration represents the status of an EZID identifier. The status is recorded as the value of the "_status"
005 * EZID metadata element.
006 */
007public enum IdentifierStatus {
008  /**
009   * A reserved identifier is only known to EZID. It may be specified only at identifier creation time.
010   */
011  RESERVED,
012  /**
013   * A public identifier will be registered with resolvers and other external services. It may be marked as unavailable.
014   */
015  PUBLIC,
016  /**
017   * An unavailable identifier is public, but the object referenced by the identifier is not available with an optional
018   * message explaining why following the status. The identifier redirects to an EZID-provied tombstone page regardless
019   * of its target URL. An unavailable identifier will be removed from any external services. It may be returned to
020   * public status.
021   */
022  UNAVAILABLE
023
024}
