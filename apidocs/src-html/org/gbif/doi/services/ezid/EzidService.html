032  /**
033   * This creates the EZID URI, and ensures that it is formatted correctly, e.g.
034   * E.g. http://ezid.cdlib.org/id/doi:10.nnnn/suffix Notice the DOI must start with the DOI access schema ("doi:").
035   * </br>
036   * The EZID URI is used to manipulate the identifier and is used as the address in the HTTP request.
037   *
038   * @param doi identifier
039   * @return EZID URI
040   */
041  public static URI ezidUri(DOI doi) {
042    if (doi != null) {
043      try {
044        return new URI(ID_ENDPOINT_URI + doi.toString());
045      } catch (URISyntaxException e) {
046        // can this happen?
047      }
048    }
049    return null;
050  }
051
052  /**
053   * This method serializes a map of metadata name/value pairs into an ANVL String value. If the
054   * map is null, or if it is empty, then it returns a null string.
055   * </br>
056   * EZID metadata dictionaries must be serialized using a subset of A Name-Value Language (ANVL) rules:
057   * -One element name/value pair per line.
058   * -Names separated from values by colons.
059   *
060   * @param metadata the Map of metadata name/value pairs
061   *
062   * @return an ANVL String or null if no metadata name/value pairs provided
063   */
064  private String serializeAsANVL(HashMap<String, String> metadata) {
065    StringBuffer buffer = new StringBuffer();
066    if (metadata != null && !metadata.isEmpty()) {
067      for (Map.Entry<String, String> entry : metadata.entrySet()) {
068        buffer.append(escape(entry.getKey()) + ": " + escape(entry.getValue()) + "\n");
069      }
070    }
071    return buffer.toString();
072  }
073
074  /**
075   * This method escapes a string to produce its ANVL escaped equivalent.
076   * </br>
077   * Care must be taken to escape structural characters that appear in element names and values, specifically,
078   * line terminators (both newlines ("\n") and carriage returns ("\r") and, in element names, colons
079   * (":"). EZID employs percent-encoding as the escaping mechanism, and thus percent signs ("%") must
080   * be escaped as well.
081   *
082   * @param str string escape
083   *
084   * @return the escaped String, or an empty string if the string was empty or null
085   */
086  private String escape(String str) {
087    if (!Strings.isNullOrEmpty(str)) {
088      return str.replace("%", "%25").replace("\n", "%0A").replace("\r", "%0D").replace(":", "%3A");
089    }
090    return "";
091  }
