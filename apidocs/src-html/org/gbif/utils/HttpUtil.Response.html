153  /**
154   * Creates a url form encoded http entity suitable for POST requests with a single given parameter
155   * encoded in utf8.
156   * 
157   * @param kvp the parameter map to encode
158   */
159  public static HttpEntity map2Entity(Map<String, String> kvp) {
160    List<NameValuePair> formparams = new ArrayList<NameValuePair>(kvp.size());
161    for (Map.Entry<String, String> entry : kvp.entrySet()) {
162      formparams.add(new BasicNameValuePair(entry.getKey(), entry.getValue()));
163    }
164    try {
165      return new UrlEncodedFormEntity(formparams, UTF_8);
166    } catch (UnsupportedEncodingException e) {
167      LOG.error("Can't encode post entity with utf8", e);
168    }
169    return null;
170  }

174   * encoded in utf8.

179  public static HttpEntity map2Entity(String key, String data) {

183      return new UrlEncodedFormEntity(formparams, UTF_8);
184    } catch (UnsupportedEncodingException e) {
185      LOG.error("Can't encode post entity with utf8", e);
186    }
187    return null;
188  }

190  /**
191   * This creates a new threadsafe, multithreaded http client with support for http and https.
192   * Default http client values are partially overriden to use UTF8 as the default charset and an explicit timeout
193   * is required for configuration.
194   * 
195   * @param timeout in milliseconds
196   * @param maxConnections maximum allowed connections in total
197   * @param maxPerRoute maximum allowed connections per route
198   */
199  public static DefaultHttpClient newMultithreadedClient(int timeout, int maxConnections, int maxPerRoute) {
200    HttpParams params = new BasicHttpParams();
201    params.setParameter(CoreProtocolPNames.HTTP_CONTENT_CHARSET, UTF_8);
202    HttpConnectionParams.setConnectionTimeout(params, timeout);
203    HttpConnectionParams.setSoTimeout(params, timeout);
204    params.setLongParameter(ClientPNames.CONN_MANAGER_TIMEOUT, timeout);
205
206    SchemeRegistry schemeRegistry = new SchemeRegistry();
207    schemeRegistry.register(new Scheme(HTTP_PROTOCOL, HTTP_PORT, PlainSocketFactory.getSocketFactory()));
208    schemeRegistry.register(new Scheme(HTTPS_PROTOCOL, HTTPS_PORT, SSLSocketFactory.getSocketFactory()));
209
210    PoolingClientConnectionManager connectionManager = new PoolingClientConnectionManager(schemeRegistry);
211    connectionManager.setMaxTotal(maxConnections);
212    connectionManager.setDefaultMaxPerRoute(maxPerRoute);
213    return new DefaultHttpClient(connectionManager, params);
214  }

217  /**
218   * Parses a RFC2616 compliant date string such as used in http headers.
219   * 
220   * @see <a href="http://tools.ietf.org/html/rfc2616#section-3.3">RFC 2616</a> specification.
221   *      example:
222   *      Wed, 21 Jul 2010 22:37:31 GMT
223   * @param rfcDate RFC2616 compliant date string
224   * @return the parsed date or null if it cannot be parsed
225   */
226  public static Date parseHeaderDate(String rfcDate) {
227    try {
228      if (rfcDate != null) {
229        // as its not thread safe we create a new instance each time
230        return new SimpleDateFormat("EEE, dd MMM yyyy HH:mm:ss Z", Locale.US).parse(rfcDate);
231      }
232    } catch (ParseException e) {
233      LOG.debug("Can't parse RFC2616 date");
234    }
235    return null;
236  }

238  public static String responseAsString(HttpResponse response) {
239    String content = null;
240    HttpEntity entity = response.getEntity();
241    if (entity != null) {
242      try {
243        content = EntityUtils.toString(entity);
244        EntityUtils.consume(entity);
245      } catch (org.apache.http.ParseException e) {
246        LOG.error("ParseException consuming http response into string", e);
247      } catch (IOException e) {
248        LOG.error("IOException consuming http response into string", e);
249      }
250    }
251    return content;
252  }

274  private HttpContext buildContext(String uri, UsernamePasswordCredentials credentials) throws URISyntaxException {
275    HttpContext authContext = new BasicHttpContext();
276    if (credentials != null) {
277      URI authUri = new URI(uri);
278      AuthScope scope = new AuthScope(authUri.getHost(), AuthScope.ANY_PORT, AuthScope.ANY_REALM);
279
280      CredentialsProvider credsProvider = new BasicCredentialsProvider();
281      credsProvider.setCredentials(scope, credentials);
282
283      authContext.setAttribute(HttpClientContext.CREDS_PROVIDER, credsProvider);
284    }
285
286    return authContext;
287  }

290   * Executes a generic DELETE request.
291   */
292  public Response delete(String url, UsernamePasswordCredentials credentials) throws IOException, URISyntaxException {
293    LOG.info("Http delete to {}", url);
294    HttpDelete delete = new HttpDelete(url);
295    HttpContext authContext = buildContext(url, credentials);
296    CloseableHttpResponse response = client.execute(delete, authContext);
297    Response result = new Response(response);
298    try {
299      HttpEntity entity = response.getEntity();
300      if (entity != null) {
301        result.content = EntityUtils.toString(entity);
302        EntityUtils.consume(entity);
303      }
304    } finally {
305      closeQuietly(response);
306    }
307    return result;
308  }

331  public StatusLine download(URL url, File downloadTo) throws IOException {
332    HttpGet get = new HttpGet(url.toString());
333
334    // execute
335    CloseableHttpResponse response = client.execute(get);
336    final StatusLine status = response.getStatusLine();
337    try {
338      // write to file only when download succeeds
339      if (success(status)) {
340        saveToFile(response, downloadTo);
341        LOG.debug("Successfully downloaded {} to {}", url, downloadTo.getAbsolutePath());
342      } else {
343        LOG.error("Downloading {} to {} failed!: {}", url, downloadTo.getAbsolutePath(), status.getStatusCode());
344      }
345    } finally {
346      closeQuietly(response);
347    }
348    return status;
349  }

351  /**
352   * @return body content if changed or null if unmodified since lastModified
353   */
354  public String downloadIfChanged(URL url, Date lastModified) throws IOException {
355    Map<String, String> header = new HashMap<String, String>(1);
356    header.put(MODIFIED_SINCE, DateFormatUtils.SMTP_DATETIME_FORMAT.format(lastModified));
357
358    try {
359      Response resp = get(url.toString(), header, null);
360      if (resp.getStatusCode() == HttpStatus.SC_NOT_MODIFIED) {
361        LOG.debug("Content not modified since last request");
362      }
363      return resp.content;
364    } catch (URISyntaxException e) {
365      LOG.error("Invalid URL provided: {}", url, e);
366    }
367    return null;
368  }

392  /**
393   * Downloads a url to a file if its modified since the date given.
394   * Updates the last modified file property to reflect the last servers modified http header.
395   * 
396   * @param downloadTo file to download to
397   * @return true if changed or false if unmodified since lastModified
398   */
399  public StatusLine downloadIfModifiedSince(final URL url, final Date lastModified, final File downloadTo) throws IOException {
400
401    HttpGet get = new HttpGet(url.toString());
402
403    // prepare conditional GET request headers
404    if (lastModified != null) {
405      // DateFormatUtils is threadsafe
406      get.addHeader(MODIFIED_SINCE, DateFormatUtils.SMTP_DATETIME_FORMAT.format(lastModified));
407      LOG.debug("Conditional GET: {}", DateFormatUtils.SMTP_DATETIME_FORMAT.format(lastModified));
408    }
409
410    // execute
411    CloseableHttpResponse response = client.execute(get);
412    final StatusLine status = response.getStatusLine();
413    try {
414      if (status.getStatusCode() == HttpStatus.SC_NOT_MODIFIED) {
415        LOG.debug("Content not modified since last request");
416
417      } else if (success(status)) {
418        // write to file only when download succeeds
419        saveToFile(response, downloadTo);
420        LOG.debug("Successfully downloaded {} to {}", url, downloadTo.getAbsolutePath());
421
422      } else {
423        LOG.error("Downloading {} to {} failed!: {}", url, downloadTo.getAbsolutePath(), status.getStatusCode());
424      }
425
426    } finally {
427      closeQuietly(response);
428    }
429    return status;
430  }

432  private void saveToFile(CloseableHttpResponse response, File downloadTo) throws IOException {
433    HttpEntity entity = response.getEntity();
434    if (entity != null) {
435      Date serverModified = null;
436      Header modHeader = response.getFirstHeader(LAST_MODIFIED);
437      if (modHeader != null) {
438        serverModified = parseHeaderDate(modHeader.getValue());
439      }
440
441      // copy stream to local file
442      FileUtils.forceMkdir(downloadTo.getParentFile());
443      OutputStream fos = new FileOutputStream(downloadTo, false);
444      try {
445        entity.writeTo(fos);
446      } finally {
447        fos.close();
448      }
449      // update last modified of file with http header date from server
450      if (serverModified != null) {
451        downloadTo.setLastModified(serverModified.getTime());
452      }
453    }
454  }

456  /**
457   * Downloads a url to a local file using conditional GET, i.e. only downloading the file again if it has been changed
458   * on the filesystem since the last download.
459   * 
460   * @param url url to download
461   * @param downloadTo file to download into and used to get the last modified date from
462   */
463  public StatusLine downloadIfModifiedSince(final URL url, final File downloadTo) throws IOException {
464    Date lastModified = null;
465    if (downloadTo.exists()) {
466      lastModified = new Date(downloadTo.lastModified());
467    }
468    return downloadIfModifiedSince(url, lastModified, downloadTo);
469  }

471  public CloseableHttpResponse executeGetWithTimeout(HttpGet get, int timeout) throws IOException {
472    HttpParams httpParams = client.getParams();
473    // keep old values to rest afterwards
474    int ct = HttpConnectionParams.getConnectionTimeout(httpParams);
475    int st = HttpConnectionParams.getSoTimeout(httpParams);
476
477    HttpConnectionParams.setConnectionTimeout(httpParams, timeout);
478    HttpConnectionParams.setSoTimeout(httpParams, timeout);
479
480    CloseableHttpResponse response = null;
481    try {
482      response = client.execute(get);
483    } finally {
484      // rest to previous values
485      HttpConnectionParams.setConnectionTimeout(httpParams, ct);
486      HttpConnectionParams.setSoTimeout(httpParams, st);
487    }
488
489    return response;
490  }

499  public Response get(String url, Map<String, String> headers, UsernamePasswordCredentials credentials)
500    throws IOException, URISyntaxException {
501    HttpGet get = new HttpGet(url);
502    // http header
503    if (headers != null) {
504      for (Map.Entry<String, String> header : headers.entrySet()) {
505        get.addHeader(StringUtils.trimToEmpty(header.getKey()), StringUtils.trimToEmpty(header.getValue()));
506      }
507    }
508    // authentication
509    HttpContext authContext = buildContext(url, credentials);
510    CloseableHttpResponse response = client.execute(get, authContext);
511    Response result = new Response(response);
512    try {
513      HttpEntity entity = response.getEntity();
514      if (entity != null) {
515        // Adding a default charset in case it is not found
516        result.content = EntityUtils.toString(entity, UTF_8);
517      }
518    } finally {
519      closeQuietly(response);
520    }
521
522    return result;
523  }

551  public Response post(String uri, Map<String, String> headers, UsernamePasswordCredentials credentials,
552    HttpEntity requestEntity) throws IOException, URISyntaxException {
553    HttpPost post = new HttpPost(uri);
554    if (headers != null) {
555      for (Map.Entry<String, String> header : headers.entrySet()) {
556        post.addHeader(StringUtils.trimToEmpty(header.getKey()), StringUtils.trimToEmpty(header.getValue()));
557      }
558    }
559    if (requestEntity != null) {
560      post.setEntity(requestEntity);
561    }
562    // authentication
563    HttpContext authContext = buildContext(uri, credentials);
564    HttpResponse response = client.execute(post, authContext);
565
566    // response
567    if (response != null) {
568      Response result = new Response(response);
569      HttpEntity respEntity = response.getEntity();
570      if (respEntity != null) {
571        result.content = EntityUtils.toString(respEntity);
572        EntityUtils.consume(respEntity);
573      }
574      return result;
575    }
576    return null;
577  }

592  public boolean verifyHost(HttpHost host) {
593    if (host != null) {
594      CloseableHttpResponse resp = null;
595      try {
596        HttpHead head = new HttpHead(host.toURI());
597        resp = client.execute(host, head);
598        return true;
599
600      } catch (Exception e) {
601        LOG.debug("Exception thrown", e);
602
603      } finally {
604        closeQuietly(resp);
605      }
606    }
607    return false;
608  }

610  private void closeQuietly(CloseableHttpResponse resp) {
611    if (resp != null) {
612      try {
613        resp.close();
614      } catch (IOException e) {
615        LOG.debug("Failed to close http response", e);
616      }
617    }
