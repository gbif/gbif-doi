024package com.kscs.util.jaxb;
025
026import java.util.Collections;
027import java.util.Map;
028
029/**
030 * Helper class acting as base class for all selectors and
031 * concrete implementation of leaf selectors.
032 * @author mirko 2014-04-04
033 */
034public class Selector<TRoot extends Selector<TRoot, ?>, TParent> {
035        public final TRoot _root;
036        public final TParent _parent;
037        protected final String _propertyName;
038        protected final boolean _include;
039
040        @SuppressWarnings("unchecked")
041        public Selector(final TRoot root, final TParent parent, final String propertyName, final boolean include) {
042                        this._root = root == null ? (TRoot) this : root;
043                        this._parent = parent;
044                        this._propertyName = propertyName;
045                        this._include = include;
046        }
047
048        public Selector(final TRoot root, final TParent parent, final String propertyName) {
049                this(root, parent, propertyName, true);
050        }
051
052        /**
053         * This is only used by builders and other implementational details
054         */
055        public Map<String, PropertyTree> buildChildren() {
056                return Collections.emptyMap();
057        }
058
059        public PropertyTree build() {
060                return this._root.init();
061        }
062
063        /**
064         * This is only used by builders and other implementational details
065         */
066        public PropertyTree init() {
067                return new PropertyTree(this._propertyName, buildChildren());
068        }
069}
