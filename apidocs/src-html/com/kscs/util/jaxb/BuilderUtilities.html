024package com.kscs.util.jaxb;
025
026import java.lang.reflect.Constructor;
027import java.lang.reflect.InvocationTargetException;
028
029/**
030 * Retrieves a dependent builder instance.
031 * @author mirko 2014-04-02
032 */
033@SuppressWarnings("unchecked")
034public final class BuilderUtilities {
035        public static <T> T getBuilder(final Class<T> declaredBuilderType, final Object instance, final Object... args) {
036                for(final Class<?> innerClass : instance.getClass().getDeclaredClasses()) {
037                        if("Builder".equals(innerClass.getSimpleName())) {
038                                for(final Constructor<?> constructor : innerClass.getConstructors()) {
039                                        if(constructor.getParameterTypes().length == args.length) {
040                                                try {
041                                                        return (T) constructor.newInstance(args);
042                                                } catch (final InstantiationException e) {
043                                                        throw new RuntimeException(e);
044                                                } catch (final IllegalAccessException e) {
045                                                        throw new RuntimeException(e);
046                                                } catch (final InvocationTargetException e) {
047                                                        throw new RuntimeException(e);
048                                                }
049                                        }
050                                }
051                        }
052                }
053                return null;
054        }
055
056}
