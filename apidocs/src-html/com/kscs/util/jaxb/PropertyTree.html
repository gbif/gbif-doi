025package com.kscs.util.jaxb;
026
027import java.util.Collections;
028import java.util.LinkedHashMap;
029import java.util.Map;
030
031/**
032 * Represents a property path for use in the clone() method
033 */
034public class PropertyTree {
035        private final Map<String, PropertyTree> children;
036        private final String propertyName;
037
038        public static final class Builder {
039                private final Map<String, Builder> children = new LinkedHashMap<String, Builder>();
040                private final Builder parent;
041                private final String propertyName;
042
043                public Builder() {
044                        this(null, null);
045                }
046
047                private Builder(final Builder parent, final String propertyName) {
048                        this.parent = parent;
049                        this.propertyName = propertyName;
050                }
051
052                public Builder with(final String propertyName) {
053                        Builder child = this.children.get(propertyName);
054                        if (child == null) {
055                                child = new Builder(this, propertyName);
056                                this.children.put(propertyName, child);
057                        }
058                        return child;
059                }
060
061
062                public PropertyTree build() {
063                        if (this.parent != null) {
064                                return this.parent.build();
065                        } else {
066                                return new PropertyTree(this.propertyName, buildChildren());
067                        }
068                }
069
070                private Map<String, PropertyTree> buildChildren() {
071                        final Map<String, PropertyTree> childProducts = new LinkedHashMap<String, PropertyTree>(this.children.size());
072                        for (final Builder childBuilder : this.children.values()) {
073                                final PropertyTree child = new PropertyTree(childBuilder.propertyName, childBuilder.buildChildren());
074                                childProducts.put(child.propertyName, child);
075                        }
076                        return Collections.unmodifiableMap(childProducts);
077                }
078
079
080                public Builder parent() {
081                        return this.parent;
082                }
083
084                public Builder root() {
085                        if (this.parent != null) {
086                                return this.parent.root();
087                        } else {
088                                return this;
089                        }
090                }
091        }
092
093        public static Builder builder() {
094                return new Builder();
095        }
096
097        public PropertyTree(final String propertyName, final Map<String, PropertyTree> children) {
098                this.propertyName = propertyName;
099                this.children = Collections.unmodifiableMap(children);
100        }
101
102        public PropertyTree get(final String propertyName) {
103                return isLeaf() ? null : this.children.get(propertyName);
104        }
105
106        public boolean isLeaf() {
107                return this.children == null || this.children.isEmpty();
108        }
109
110        public String propertyName() {
111                return this.propertyName;
112        }
113
114}
