53    /**
54     * This method serializes a map of metadata name/value pairs into an ANVL String value. If the
55     * map is null, or if it is empty, then it returns a null string.
56     * </br>
57     * EZID metadata dictionaries must be serialized using a subset of A Name-Value Language (ANVL) rules:
58     * -One element name/value pair per line.
59     * -Names separated from values by colons.
60     *
61     * @param metadata the Map of metadata name/value pairs
62     *
63     * @return an ANVL String or null if no metadata name/value pairs provided
64     */
65    private String serializeAsANVL(HashMap<String, String> metadata) {
66      StringBuffer buffer = new StringBuffer();
67      if (metadata != null && !metadata.isEmpty()) {
68        for (Map.Entry<String, String> entry : metadata.entrySet()) {
69          buffer.append(escape(entry.getKey()) + ": " + escape(entry.getValue()) + "\n");
70        }
71      }
72      return buffer.toString();
73    }

75    /**
76     * This method escapes a string to produce its ANVL escaped equivalent.
77     * </br>
78     * Care must be taken to escape structural characters that appear in element names and values, specifically,
79     * line terminators (both newlines ("\n") and carriage returns ("\r") and, in element names, colons
80     * (":"). EZID employs percent-encoding as the escaping mechanism, and thus percent signs ("%") must
81     * be escaped as well.
82     *
83     * @param str string escape
84     *
85     * @return the escaped String, or an empty string if the string was empty or null
86     */
87    private String escape(String str) {
88      if (!Strings.isNullOrEmpty(str)) {
89        return str.replace("%", "%25").replace("\n", "%0A").replace("\r", "%0D").replace(":", "%3A");
90      }
91      return "";
92    }
