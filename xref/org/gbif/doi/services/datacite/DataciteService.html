51      // a random DOI might exist already so we need to try to reserve one until we find a good one.
52      // to avoid endless loop do a max of 100 tries though, we should never ever reach that
53      for (int x=1; x<100; x++) {
54        DOI doi = random(prefix);
55        try {
56          reserve(doi, metadata);
57          return doi;
58        } catch (DoiExistsException e) {
59          // this doi did exist already, mint another one
60        }
61      }
62      LOG.error("Failed to generate a new, random DOI for prefix {}. We have tried 100 random suffices.", prefix);
63      throw new DoiException("Failed to generate a new, random DOI for prefix " + prefix);
64    }
