40    public DataciteService(CloseableHttpClient httpClient, ServiceConfig cfg) {

50    @Override
51    @Nullable
52    public URI get(DOI doi) throws DoiException {
53      Preconditions.checkNotNull(doi);
54      try {
55        String content = get(doiUri(doiWs, doi));
56        if (!Strings.isNullOrEmpty(content)) {
57          return URI.create(content);
58        }
59      } catch (IllegalArgumentException e) {
60        LOG.debug("Invalid DOI URL", e.getMessage());
61      } catch (DoiHttpException e) {
62        if (Math.round( e.getStatus() / 100.0) == 4) {
63          throw e;
64        }
65        LOG.debug("Non existing DOI", e.getMessage());
66      } catch (DoiException e) {
67        LOG.debug("Non existing DOI", e.getMessage());
68      }
69      return null;
70    }

73     * @return true if the DOI is reserved or registered

75    private boolean exists(DOI doi) throws DoiException {
76      try {
77        String content = get(doiUri(metadataWs, doi));

79  
80        return false;
81      }
82      return true;

85    /**
86     * Checks of a DOI is publicly registered already.
87     * Registered but deleted DOIs are still considered registered and return true!
88     * @return true if the DOI is registered
89     */
90    private boolean isRegistered(DOI doi) throws DoiException {
91      try {
92        URI url = get(doi);
93        if (url != null && !TEST_404_TARGET.equals(url)) {
94          return true;
95        }
96      } catch (DoiHttpException e) {
97      }
98      return false;
99    }

101   @Override
102   public void reserve(DOI doi, DataCiteMetadata metadata) throws DoiException {
103     Preconditions.checkNotNull(doi);
104     Preconditions.checkNotNull(metadata);
105     if (exists(doi)) {
106       throw new DoiException("DOI exists already");
107     }
108     LOG.info("Reserving {}", doi);
109     updateMetadataIdentifier(doi, metadata);
110     post(doi, metadataWs, metadata);
111   }

113   @Override
114   public DOI reserveRandom(String prefix, int length, DataCiteMetadata metadata) throws DoiException {
115     Preconditions.checkArgument(length > 2, "The suffix length must be larger than 2");
116     Preconditions.checkArgument(prefix.startsWith("10."), "The DOI prefix must begin with 10");
117 
118     // a random DOI might exist already so we need to try to reserve one until we find a good one.
119     // to avoid endless loop do a max of 100 tries though, we should never ever reach that
120     for (int x=1; x<100; x++) {
121       DOI doi = random(prefix, length);
122       if (!exists(doi)) {
123         reserve(doi, metadata);
124         return doi;
125       }
126     }
127     LOG.error("Failed to generate a new, random DOI for prefix {}. We have tried 100 random suffices.", prefix);
128     throw new DoiException("Failed to generate a new, random DOI for prefix " + prefix);
129   }

131   @Override
132   public void register(DOI doi, URI target) throws DoiException {
133     Preconditions.checkNotNull(doi);
134     Preconditions.checkNotNull(target);
135     if (isRegistered(doi)) {
136       throw new DoiException("DOI was already registered");
137     }
138     LOG.info("Registering {}", doi);
139     authCall(buildDoiUrlPost(doi, target));
140   }

160   @Override
161   public void update(DOI doi, URI target) throws DoiException {
162     Preconditions.checkNotNull(doi);
163     Preconditions.checkNotNull(target);
164     if (!isRegistered(doi)) {
165       throw new DoiException("DOI was not registered yet");
166     }
167     LOG.info("Updating target URL for {}", doi);
168     authCall(buildDoiUrlPost(doi, target));
169   }

171   private HttpPost buildDoiUrlPost(DOI doi, URI target) {
172     HttpPost post = new HttpPost(doiWs);
173     String body = String.format("doi=%s\nurl=%s", doi.getDoiName(), target);
174     HttpEntity entity = new StringEntity(body, PALINT_TEXT_UTF8);
175     post.setEntity(entity);
176     return post;
177   }
