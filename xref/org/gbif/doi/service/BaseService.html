63    /**
64     * Produces a validated xml representation of the data cite metadata.
65     * The DOI inside the metadata is overwritten to always represent the given DOI
66     * and the generated XML is validated according to the datacite xsd schema.
67     * @return xml string
68     * @throws InvalidMetadataException, DoiException
69     */
70    @VisibleForTesting
71    protected String toXml(DOI doi, DataCiteMetadata data) throws InvalidMetadataException {
72      // make sure we use the right doi inside the metadata
73      data.setIdentifier(DataCiteMetadata.Identifier.builder()
74          .withValue(doi.getDoiName())
75          .withIdentifierType("DOI")
76          .build()
77      );
78      // (un)marshaller are not thread safe and need to be created on each authCall
79      StringWriter writer = new StringWriter();
80      try {
81        Marshaller m = context.createMarshaller();
82        m.setProperty(Marshaller.JAXB_FORMATTED_OUTPUT, Boolean.TRUE);
83        m.setProperty(Marshaller.JAXB_SCHEMA_LOCATION, DATACITE_SCHEMA_LOCATION);
84  
85        m.marshal(data, writer);
86        writer.flush();
87      } catch (JAXBException e) {
88        throw new InvalidMetadataException(doi, e);
89      }
90  
91      final String xml = writer.toString();
92      // validate the xml before we send it to datacite
93      try {
94        DataciteValidator.validateMetadata(xml);
95        LOG.debug("Metadata XML passed validation", doi);
96  
97      } catch (SAXException e) {
98        throw new InvalidMetadataException(doi, e);
99  
100     } catch (IOException e) {
101       // cant happen, we had a string in memory already!
102       throw new IllegalStateException(e);
103     }
104     return xml;
105   }

107   /**
108    * Does an authenticated call and throws a DoiHttpException for nun successful responses other than 2xx.
109    * If successful return the entities body as a string.
110    * @throws DoiException
111    */
112   protected String authCall(HttpUriRequest req) throws DoiException {
113     CloseableHttpResponse resp = null;
114     try {
115       // authentication
116       resp = httpClient.execute(req, buildAuthContext());
117       // Everything but HTTP 2xx is an error
118       if (!isSuccess(resp)) {
119         // read response body and use as error message
120         String msg = EntityUtils.toString(resp.getEntity(), Charsets.UTF_8);
121         LOG.debug("Received HTTP {} {}: {}", resp.getStatusLine().getStatusCode(), resp.getStatusLine().getReasonPhrase(), msg);
122         closeSilently(resp);
123         throw new DoiHttpException(resp.getStatusLine().getStatusCode(), msg);
124       }
125       return EntityUtils.toString(resp.getEntity(), Charsets.UTF_8);
126 
127     } catch (DoiException e) {
128       throw e;
129     } catch (Throwable e) {
130       throw new DoiException(e);
131     } finally {
132       closeSilently(resp);
133     }
134   }
135 
136   private static void closeSilently(CloseableHttpResponse resp) {
137     if (resp != null) {
138       try {
139         resp.close();
140       } catch (IOException e) {
141         LOG.warn("Could not close http connection", e);
142         // ignore, we're done the request was executed
143       }
144     }
145   }

168   @Override
169   public DOI reserveRandom(String prefix, String shoulder, int length, DataCiteMetadata metadata) throws DoiException {
170     Preconditions.checkArgument(length > 2, "The suffix length must be larger than 2");
171     Preconditions.checkArgument(prefix.startsWith("10."), "The DOI prefix must begin with 10");
172 
173     // a random DOI might exist already so we need to try to reserve one until we find a good one.
174     // to avoid endless loop do a max of 100 tries though, we should never ever reach that
175     for (int x=1; x<100; x++) {
176       DOI doi = random(prefix, shoulder, length);
177       try {
178         reserve(doi, metadata);
179         return doi;
180       } catch (DoiExistsException e) {
181         LOG.debug("Randoom DOI existed. Try another one", e.getMessage());
182       }
183     }
184     LOG.error("Failed to generate a new, random DOI for prefix {}. We have tried 100 random suffices.", prefix);
185     throw new DoiException("Failed to generate a new, random DOI for prefix " + prefix);
186   }

198   protected static DOI random(String prefix, String shoulder, int length) {
199     String suffix = Strings.nullToEmpty(shoulder) + RandomStringUtils.randomAlphanumeric(length);
