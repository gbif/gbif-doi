28  public abstract class BaseService implements DoiService {
29  
30    private static final Logger LOG = LoggerFactory.getLogger(BaseService.class);
31    protected static final ContentType PLAIN_TEXT_UTF8 = ContentType.create("text/plain", Charsets.UTF_8);
32  
33    private final CloseableHttpClient httpClient;
34    private final CredentialsProvider credsProvider;
35  
36    protected BaseService(CloseableHttpClient httpClient, ServiceConfig cfg) {
37      this.httpClient = httpClient;
38      LOG.info("Setup DOI service using account {}", cfg.getUsername());
39      credsProvider = new BasicCredentialsProvider();
40      credsProvider.setCredentials(AuthScope.ANY, new UsernamePasswordCredentials(cfg.getUsername(), cfg.getPassword()));
41    }

43    /**
44     * Does an authenticated call and throws a DoiHttpException for nun successful responses other than 2xx.
45     * If successful return the entities body as a string.
46     * @throws DoiException
47     */
48    protected String authCall(HttpUriRequest req) throws DoiException {
49      CloseableHttpResponse resp = null;
50      try {
51        // authentication
52        resp = httpClient.execute(req, buildAuthContext());
53        // Everything but HTTP 2xx is an error
54        if (!isSuccess(resp)) {
55          // read response body and use as error message
56          String msg = EntityUtils.toString(resp.getEntity(), Charsets.UTF_8);
57          LOG.debug("Received HTTP {} {}: {}", resp.getStatusLine().getStatusCode(), resp.getStatusLine().getReasonPhrase(), msg);
58          closeSilently(resp);
59          throw new DoiHttpException(resp.getStatusLine().getStatusCode(), msg);
60        }
61        return EntityUtils.toString(resp.getEntity(), Charsets.UTF_8);
62  
63      } catch (DoiException e) {
64        throw e;
65      } catch (Throwable e) {
66        throw new DoiException(e);
67      } finally {
68        closeSilently(resp);
69      }
70    }
71  
72    private static void closeSilently(CloseableHttpResponse resp) {
73      if (resp != null) {
74        try {
75          resp.close();
76        } catch (IOException e) {
77          LOG.warn("Could not close http connection", e);
78          // ignore, we're done the request was executed
79        }
80      }
81    }
82  
83    /**
84     * Does an authenticated http GET and returns the full body entity as a String
85     */
86    protected String get(URI uri) throws DoiException {
87      return authCall(new HttpGet(uri));
88    }
89  
90    /**
91     * Does an authenticated http DELETE and returns the full body entity as a String
92     */
93    protected String delete(URI uri) throws DoiException {
94      return authCall(new HttpDelete(uri));
95    }
96  
97    protected static boolean isSuccess(HttpResponse resp) {
98      if (Math.round( resp.getStatusLine().getStatusCode() / 100.0) == 2) {
99        return true;
100     }
101     return false;
102   }

104   private HttpContext buildAuthContext() {
105     HttpContext authContext = new BasicHttpContext();
106     authContext.setAttribute(HttpClientContext.CREDS_PROVIDER, credsProvider);
107     return authContext;
108   }
