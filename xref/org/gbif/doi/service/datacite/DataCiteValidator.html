44    /**
45     * Produces a validated xml representation of the data cite metadata.
46     * The DOI inside the metadata is overwritten to always represent the given DOI
47     * and the generated XML is validated according to the datacite xsd schema.
48     * @return xml string
49     * @throws InvalidMetadataException if the generated xml failed validation, e.g. missing mandatory elements
50     */
51    public static String toXml(DOI doi, DataCiteMetadata data) throws InvalidMetadataException {
52      // make sure we use the right doi inside the metadata
53      data.setIdentifier(DataCiteMetadata.Identifier.builder()
54          .withValue(doi.getDoiName())
55          .withIdentifierType("DOI")
56          .build()
57      );
58      // (un)marshaller are not thread safe and need to be created on each authCall
59      StringWriter writer = new StringWriter();
60      try {
61        Marshaller m = context.createMarshaller();
62        m.setProperty(Marshaller.JAXB_FORMATTED_OUTPUT, Boolean.TRUE);
63        m.setProperty(Marshaller.JAXB_SCHEMA_LOCATION, DATACITE_SCHEMA_LOCATION);
64  
65        m.marshal(data, writer);
66        writer.flush();
67      } catch (JAXBException e) {
68        throw new InvalidMetadataException(e);
69      }
70  
71      final String xml = writer.toString();
72      // validate the xml before we return it
73      validateMetadata(xml);
74      LOG.debug("Metadata XML passed validation", doi);
75      return xml;
76    }
77  
78    public static DataCiteMetadata fromXml(String xml) throws JAXBException {
79      return fromXml(new ByteArrayInputStream(xml.getBytes(Charsets.UTF_8)));
80    }

82    public static DataCiteMetadata fromXml(InputStream in) throws JAXBException {
83      // (un)marshaller are not thread safe and need to be created on each call
84      Unmarshaller m = context.createUnmarshaller();
85      return (DataCiteMetadata) m.unmarshal(in);
86    }

99    public static void validateMetadata(Source source) throws InvalidMetadataException {
100     try {
101       getValidator().validate(source);
102       LOG.debug("Metadata XML passed validation");
103 
104     } catch (SAXException e) {
105       throw new InvalidMetadataException(e);
106 
107     } catch (IOException e) {
108       // cant happen, we had a string in memory already!
109       throw new IllegalStateException(e);
110     }
111   }

113   private static Validator getValidator() throws IOException, SAXException {
114     if (DATACITE_VALIDATOR == null) {
115       // define the type of schema - we use W3C:
116       String schemaLang = "http://www.w3.org/2001/XMLSchema";
117       // resolve validation driver:
118       SchemaFactory factory = SchemaFactory.newInstance(schemaLang);
119       // create schema by reading it from an URL:
120       //File f = FileUtils.getClasspathFile("datacite/metadata.xsd");
121       Schema schema = factory.newSchema(new StreamSource("http://schema.datacite.org/meta/kernel-3/metadata.xsd"));
122       DATACITE_VALIDATOR = schema.newValidator();
123     }
124     return DATACITE_VALIDATOR;
125   }
