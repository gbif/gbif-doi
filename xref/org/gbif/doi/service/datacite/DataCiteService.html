49    /**
50     * Detecting the status might need 2 calls.
51     * A reserved DOI returns metadata and either the test or NULL target URL.
52     * An active registered DOI returns metadata and a target URI
53     * A deleted but registered DOI returns a http 410 for the metadata and still the target URI.
54     */
55    @Override
56    public DoiData resolve(DOI doi) throws DoiException {
57      Preconditions.checkNotNull(doi);
58      try {
59  
60        final URI target = getTargetUrl(doi);
61        try {
62          get(metadataUri(doi));
63          return target == null ? new DoiData(DoiStatus.RESERVED, null) : new DoiData(DoiStatus.REGISTERED, target);
64  
65        } catch (DoiHttpException e) {
66          // we only see a 410 for the metadata when a registered DOI was deleted
67          if (e.getStatus() == 410) {
68            return new DoiData(DoiStatus.DELETED, target);
69          }
70        }
71  
72      } catch (DoiException e) {
73        LOG.debug("Unknown error detecting DOI status", e.getMessage());
74      }
75      return null;
76    }

78    private URI getTargetUrl(DOI doi) throws DoiHttpException {
79      // get target URI
80      String content = null;
81      try {
82        content = get(doiUri(doi));
83        if (!Strings.isNullOrEmpty(content)) {
84          URI target = URI.create(content);
85          if (!TEST_404_TARGET.equals(target)) {
86            return target;
87          }
88        }
89      } catch (IllegalArgumentException e) {
90        LOG.debug("DataCite returned an invalid target URL {}", content, e.getMessage());
91      } catch (DoiHttpException e) {
92        if (Math.round( e.getStatus() / 100.0) == 4) {
93          throw e;
94        }
95        LOG.debug("Non existing DOI", e.getMessage());
96      } catch (DoiException e) {
97        LOG.debug("Non existing DOI", e.getMessage());
98      }
99      return null;
100   }

146   private void registerWithoutValidation(DOI doi, URI target, String metadata) throws DoiException {
147     Preconditions.checkNotNull(doi);
148     Preconditions.checkNotNull(target);
149     Preconditions.checkNotNull(metadata);
150     DoiData status = resolve(doi);
151     if (status != null && DoiStatus.REGISTERED == status.getStatus()) {
152       throw new DoiExistsException(doi);
153     }
154     post(doi, metadataWs, metadata);
155     authCall(buildDoiUrlPost(doi, target));
156     if (status != null && DoiStatus.DELETED == status.getStatus()) {
157       LOG.info("Re-registered {}", doi);
158     } else {
159       LOG.info("Registered {}", doi);
160     }
161   }

182   private void updateWithoutValidation(DOI doi, String metadata) throws DoiException {
183     Preconditions.checkNotNull(doi);

187     if (status == null || DoiStatus.REGISTERED != status.getStatus()) {

207   private HttpPost buildDoiUrlPost(DOI doi, URI target) {
208     HttpPost post = new HttpPost(doiWs);
209     String body = String.format("doi=%s\nurl=%s", doi.getDoiName(), target);
210     HttpEntity entity = new StringEntity(body, PLAIN_TEXT_UTF8);
211     post.setEntity(entity);
212     return post;
213   }

233   /**

236    * @param uri  to issue request against
237    * @param metadata request body content to be serialized as XML with jackson

239    * @throws org.gbif.doi.service.DoiException in case anything goes wrong during the request.

241   private void post(DOI doi, URI uri, String metadata) throws DoiException {
242     LOG.debug("POST: {}", uri);
243     HttpPost req = new HttpPost(uri);
244     // body
245     try {
246       HttpEntity entity = new StringEntity(metadata, APPLICATION_XML_UTF8);
247       req.setEntity(entity);
