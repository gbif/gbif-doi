17  /**
18   * ANVL (A Name-Value Language) is a simple record syntax based on email headers.
19   * An ANVL record is a sequence of data elements ending in a blank line.
20   * An element consists of a label, a colon, and an optional value, and a long value may be folded (continued)
21   * onto the next line by inserting a newline and indenting the next line.
22   *
23   * ANVL is used to excahnge metadata with EZID.
24   * This utils class contains methods to (de)serialize them from a java map to a plain string.
25   *
26   * @see <a href="https://wiki.ucop.edu/display/Curation/Anvl">ANVL specs</a>
27   * @see <a href="http://ezid.cdlib.org/doc/apidoc.html#request-response-bodies">EZID API</a>
28   */

55    /**
56     * This method serializes a map of metadata name/value pairs into an ANVL String value. If the
57     * map is null, or if it is empty, then it returns a null string.
58     * </br>
59     * EZID metadata dictionaries must be serialized using a subset of A Name-Value Language (ANVL) rules:
60     * -One element name/value pair per line.
61     * -Names separated from values by colons.
62     *
63     * @see <a href="http://ezid.cdlib.org/doc/apidoc.html#request-response-bodies">EZID API</a>
64     * @param metadata the Map of metadata name/value pairs
65     *
66     * @return an ANVL String or null if no metadata name/value pairs provided
67     */
68    public static String write(Map<String, String> metadata) {
69      StringBuffer buffer = new StringBuffer();
70      if (metadata != null) {
71        for (Map.Entry<String, String> entry : metadata.entrySet()) {
72          buffer.append(encode(entry.getKey()) + ": " + encode(entry.getValue()) + "\n");
73        }
74      }
75      return buffer.toString();
76    }
77  
78    private static String encode(String s) {
79      return s.replace("%", "%25")
80        .replace("\n", "%0A")
81        .replace("\r", "%0D")
82        .replace(":", "%3A");
83    }
84  
85    private static String decode (String s) {
86      StringBuffer b = new StringBuffer();
87      int i;
88      while ((i = s.indexOf("%")) >= 0) {
89        b.append(s.substring(0, i));
90        b.append((char)Integer.parseInt(s.substring(i+1, i+3), 16));
91        s = s.substring(i+3);
92      }
93      b.append(s);
94      return b.toString();
95    }
96  
97    private static String[] parseAnvlLine (String line) {
98      String[] kv = line.split(":", 2);
99      kv[0] = decode(kv[0]).trim();
100     kv[1] = decode(kv[1]).trim();
101     return kv;
102   }

115   public static Map<String, String> read(String anvl) {
116     try {
117       return read(new BufferedReader(new StringReader(anvl)));
118     } catch (IOException e) {
119       // cant happen so we hide the IOException as a runtime exception
120       // we have a string wrapped reader
121       throw new IllegalStateException(e);
122     }
123   }

125   private static Map<String, String> read(BufferedReader br) throws IOException {
126     HashMap<String, String> d = new HashMap<String, String>();
127     String l;
128     while ((l = br.readLine()) != null) {
129       String[] kv = parseAnvlLine(l);
130       d.put(kv[0], kv[1]);
131     }
132     return d;
133   }

148   public static class Builder {
149 
150     private Map<String, String> map = new HashMap<String, String>();
151 
152     public Builder target(URI target) {
153       if (target == null) {
154         this.map.put(AnvlUtils.TARGET, "");
155       } else {
156         this.map.put(AnvlUtils.TARGET, target.toString());
157       }
158       return this;
159     }
160 
161     public Builder datacite(String datacite) {
162       this.map.put(AnvlUtils.DATACITE, datacite);
163       return this;
164     }
165 
166     public Builder status(DoiStatus status) {
167       this.map.put(AnvlUtils.STATUS, status.getEzid());
168       return this;
169     }
170 
171     public Builder entry(String key, String value) {
172       this.map.put(key, value);
173       return this;
174     }

178     }
