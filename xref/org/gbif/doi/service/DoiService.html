1   package org.gbif.doi.service;
2   
3   import org.gbif.api.model.common.DOI;
4   import org.gbif.api.model.common.DoiData;
5   import org.gbif.doi.metadata.datacite.DataCiteMetadata;
6   
7   import java.net.URI;
8   import javax.annotation.Nullable;
9   
10  public interface DoiService {
11  
12    /**
13     * Resolves the registered identifier to its status and target URL.
14     * @param doi the identifier to resolve
15     * @return the status object with the target URL the DOI is backed by
16     *         or null if DOI does not exist at all
17     *
18     * @throws DoiException if the operation failed for any reason
19     */
20    @Nullable
21    DoiData resolve(DOI doi) throws DoiException;
22  
23    /**
24     * Reserves a new identifier and defines its initial metadata.
25     * Reserved ids are not known to resolvers yet and can still be fully deleted.
26     *
27     * @param doi the identifier to reserve
28     * @param metadata the metadata to be associated with the doi given as a valid DataCite v3.1 XML document.
29     *                 The identifier inside the metadata will be overwritten by the doi parameter given
30     *
31     * @throws DoiExistsException if the DOI existed already regardless of its status
32     * @throws DoiException if the operation failed for any reason
33     */
34    void reserve(DOI doi, String metadata) throws DoiException;
35  
36    /**
37     * Reserves a new identifier and defines its initial metadata.
38     * Reserved ids are not known to resolvers yet and can still be fully deleted.
39     *
40     * @param doi the identifier to reserve
41     * @param metadata the metadata to be associated with the doi. The identifier inside the metadata will be overwritten
42     *                 by the doi parameter given
43     *
44     * @throws DoiExistsException if the DOI existed already regardless of its status
45     * @throws DoiException if the operation failed for any reason
46     */
47    void reserve(DOI doi, DataCiteMetadata metadata) throws DoiExistsException, DoiException;
48  
49    /**
50     * Registers an identifier that is either brand new, has been reserved or is currently marked as deleted.
51     * It assigns the latest metadata and a URL for resolution.
52     * This causes the DOI to be publicly registered with resolvers and other external services.
53     * <br/>
54     * DataCite and EZID restrict target URL domains and the target URL given MUST have a domain
55     * matching your account permissions.
56     * <br/>
57     * You can also call this method to re-register previously deleted, registered DOIs
58     * that are currently marked as inactive (DataCite) or unavailable (EZID). Because the API of these 2 implementations
59     * differ in how to re-register deleted DOIs (DataCite requires new metadata, EZID needs a new terget URL),
60     * we have to take both the metadata and the target URL as params.
61     *
62     * @param doi the identifier to register
63     * @param target the URL the DOI should resolve to
64     * @param metadata the metadata to be associated with the doi given as a valid DataCite v3.1 XML document.
65     *                 The identifier inside the metadata will be overwritten by the explicitly given DOI
66     *
67     * @throws DoiException if the operation failed for any reason
68     * @throws DoiExistsException if the DOI was already registered
69     */
70    void register(DOI doi, URI target, String metadata) throws DoiException;
71  
72    /**
73     * Registers an identifier that is either brand new, has been reserved or is currently marked as deleted.
74     * It assigns the latest metadata and a URL for resolution.
75     * This causes the DOI to be publicly registered with resolvers and other external services.
76     * <br/>
77     * DataCite and EZID restrict target URL domains and the target URL given MUST have a domain
78     * matching your account permissions.
79     * <br/>
80     * You can also call this method to re-register previously deleted, registered DOIs
81     * that are currently marked as inactive (DataCite) or unavailable (EZID). Because the API of these 2 implementations
82     * differ in how to re-register deleted DOIs (DataCite requires new metadata, EZID needs a new terget URL),
83     * we have to take both the metadata and the target URL as params.
84     *
85     * @param doi the identifier to register
86     * @param target the URL the DOI should resolve to
87     * @param metadata the metadata to be associated with the doi. The identifier inside the metadata will be overwritten
88     *                 by the explicitly given DOI
89     *
90     * @throws DoiException if the operation failed for any reason
91     * @throws DoiExistsException if the DOI was already registered
92     */
93    void register(DOI doi, URI target, DataCiteMetadata metadata) throws DoiExistsException, DoiException;
94  
95    /**
96     * Tries to delete an identifier. If the DOI has only been reserved it will be fully deleted,
97     * if it was registered before it cannot be deleted as DOIs are permanent identifiers.
98     * In DataCite a DOI will be marked as "inactive" though, in EZID as "unavailable".
99     * You can re-register a deleted DOI again if needed.
100    *
101    * @param doi the identifier to delete
102    * @return true if the reserved DOI was fully deleted, false if it was only marked as deleted
103    *
104    * @throws DoiException if the operation failed for any reason
105    */
106   boolean delete(DOI doi) throws DoiException;
107 
108   /**
109    * Updates the identifier metadata. This method must be called every time the object or metadata referenced by
110    * the identifier changes (e.g. a dataset gets republished, a dataset is replaced by a new major version, etc).
111    *
112    * @param doi the identifier of metadata to update
113    * @param metadata the datacite metadata given as a valid DataCite v3.1 XML document
114    *
115    * @throws DoiException if the operation failed for any reason
116    */
117   void update(DOI doi, String metadata) throws DoiException;
118 
119   /**
120    * Updates the identifier metadata. This method must be called every time the object or metadata referenced by
121    * the identifier changes (e.g. a dataset gets republished, a dataset is replaced by a new major version, etc).
122    *
123    * @param doi the identifier of metadata to update
124    * @param metadata the datacite metadata
125    *
126    * @throws DoiException if the operation failed for any reason
127    */
128   void update(DOI doi, DataCiteMetadata metadata) throws DoiException;
129 
130   /**
131    * Updates the registered identifier's target URL.
132    * DataCite and EZID restrict target URL domains and the target URL given MUST have a domain
133    * matching your account permissions.
134    *
135    * @param doi the identifier of metadata to update
136    * @param target the new URL the DOI should resolve to
137    *
138    * @throws DoiException if the operation failed for any reason
139    */
140   void update(DOI doi, URI target) throws DoiException;
141 }
