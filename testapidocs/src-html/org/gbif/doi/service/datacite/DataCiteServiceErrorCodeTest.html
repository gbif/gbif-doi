001package org.gbif.doi.service.datacite;

003import org.gbif.api.model.common.DOI;
004import org.gbif.api.model.common.DoiData;
005import org.gbif.api.model.common.DoiStatus;

007import org.gbif.doi.service.DoiHttpException;
008import org.gbif.doi.service.ServiceConfig;
009import org.gbif.utils.file.FileUtils;
010
011import java.io.File;
012import java.io.IOException;
013import java.io.InputStream;
014import java.net.URI;
015
016import com.fasterxml.jackson.databind.ObjectMapper;
017import com.fasterxml.jackson.dataformat.yaml.YAMLFactory;
018import org.junit.Test;
019
020import static junit.framework.Assert.assertEquals;
021import static org.junit.Assert.assertTrue;
022import static org.mockito.Mockito.spy;
023import static org.mockito.Mockito.when;
024
025/**
026 *
027 * Test behavior of DataCiteService when the underlying HttpClient returns Error codes.
028 *
029 */
030public class DataCiteServiceErrorCodeTest {
031
032  public final static URI TEST_TARGET = URI.create("http://www.gbif.org/datasets");
033  private File TEST_METADATA = FileUtils.getClasspathFile("metadata/minimal.xml");
034  private DOI TEST_DOI = new DOI(DOI.TEST_PREFIX, "test");
035
036  private int REQUEST_ENTITY_TOO_LARGE = 413;
037
038  private DataCiteService buildDataCiteService(){
039    try {
040      ObjectMapper mapper = new ObjectMapper(new YAMLFactory());
041      InputStream dc = FileUtils.classpathStream("datacite.yaml");
042      ServiceConfig cfg = mapper.readValue(dc, ServiceConfig.class);
043      return new DataCiteService(new MockHttpClient(REQUEST_ENTITY_TOO_LARGE), cfg);
044    } catch (IOException e) {
045      throw new IllegalStateException(e);
046    }
047  }
048
049  @Test
050  public void testRequestEntityTooLarge() throws Exception {
051
052    boolean doiHttpExceptionCaught = false;
053    DataCiteService service = buildDataCiteService();
054    service = spy(service);
055
056    // in order to test the HTTP code REQUEST_ENTITY_TOO_LARGE we need to mock the resolve(DOI) method.
057    try {
058      when(service.resolve(TEST_DOI)).thenReturn(new DoiData(DoiStatus.REGISTERED, TEST_TARGET));
059      service.update(TEST_DOI, org.apache.commons.io.FileUtils.readFileToString(TEST_METADATA));
060    }
061    catch (DoiHttpException doiHttpEx){
062      doiHttpExceptionCaught = true;
063      assertEquals(REQUEST_ENTITY_TOO_LARGE, doiHttpEx.getStatus());
064    }
065
066    assertTrue("When HttpClient triggers a 413, DoiHttpException is thrown ", doiHttpExceptionCaught);
067  }
068
069}
