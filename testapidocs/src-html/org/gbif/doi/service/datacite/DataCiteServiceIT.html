023
024public class DataCiteServiceIT extends DoiServiceIT {
025  private static DataCiteService service;
026
027  // test prefix for DataCite, behaving like normal DOIs and occasionally purged
028  public DataCiteServiceIT() {
029    super("10.5072", "gbif.", setup());
030  }
031
032  private static DoiService setup() {
033    try {
034      ObjectMapper mapper = new ObjectMapper(new YAMLFactory());
035      InputStream dc = FileUtils.classpathStream("datacite.yaml");
036      ServiceConfig cfg = mapper.readValue(dc, ServiceConfig.class);
037      System.out.println(cfg);
038      service = new DataCiteService(HttpUtil.newMultithreadedClient(10000, 3, 2), cfg);
039      return service;
040    } catch (IOException e) {
041      throw new IllegalStateException(e);
042    }
043  }
044
045  /**
046   * DataCite does only allow you to read your own identifiers.
047   * This is different how EZID handles it, hence we put the test here and not in the base class.
048   */
049  @Test
050  public void testGetForeign() throws Exception {
051    // this DOI does not belong to GBIF and we cannot read it...
052    assertNull(service.resolve(FOREIGN_DOI));
053  }
054
055  @Test
056  @Ignore("This is only to manually test the real prefix behavior. Only use the test prefix for regular tests")
057  public void testRealGBIFPrefix() throws Exception {
058
059    final DOI doi = new DOI(DOI.GBIF_PREFIX, "test."+System.nanoTime());
060    System.out.println(doi);
061    DoiData status = service.resolve(doi);
062    System.out.println(status);
063
064    DataCiteMetadata meta = DataCiteMetadataTest.testMetadata(doi, "reserve test");
065    service.reserve(doi, meta);
066    status = service.resolve(doi);
067    System.out.println(status);
068
069    service.register(doi, TEST_TARGET, meta);
070    status = service.resolve(doi);
071    System.out.println(status);
072  }
073
074  @Test
075  public void testDataCiteUri() throws Exception {
076    assertEquals(URI.create("https://mds.datacite.org/doi/10.1234/tng7y"), service.doiUri(new DOI("10.1234/TNG7Y")));
077    assertEquals(URI.create("https://mds.datacite.org/metadata/10.1234/tng7y"), service.metadataUri(new DOI("10.1234/TNG7Y")));
078  }
079}
