001package org.gbif.doi.service.datacite;
002
003import org.gbif.doi.metadata.datacite.DataCiteMetadata;
004import org.gbif.doi.metadata.datacite.DataCiteMetadataTest;
005import org.gbif.utils.file.FileUtils;
006
007import java.io.StringWriter;
008import javax.xml.bind.JAXBContext;
009import javax.xml.bind.JAXBException;
010import javax.xml.bind.Marshaller;
011
012import org.junit.Test;
013
014public class DataCiteValidatorTest {
015  // we only use this to serialize XML
016  private static final JAXBContext context;
017  static {
018    try {
019      context = JAXBContext.newInstance(DataCiteMetadata.class);
020    } catch (JAXBException e) {
021      throw new IllegalStateException("Fail to setup JAXB", e);
022    }
023  }
024
025
026  public static String toXml(DataCiteMetadata data) throws JAXBException {
027    // (un)marshaller are not thread safe and need to be created on each postOrPut
028    Marshaller m = context.createMarshaller();
029    m.setProperty(Marshaller.JAXB_FORMATTED_OUTPUT, Boolean.TRUE);
030
031    StringWriter writer = new StringWriter();
032    m.marshal(data, writer);
033    writer.flush();
034    return writer.toString();
035  }
036
037  @Test
038  public void testValidateMetadata() throws Exception {
039    DataCiteValidator.validateMetadata(FileUtils.classpathStream("metadata/datacite-example-full-v3.1.xml"));
040    DataCiteValidator.validateMetadata(FileUtils.classpathStream("metadata/minimal.xml"));
041  }
042
043  @Test
044  public void testValidateMetadataBean() throws Exception {
045    DataCiteValidator.validateMetadata(toXml(DataCiteMetadataTest.testMetadata()));
046  }
047}
