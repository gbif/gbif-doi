001package org.gbif.doi.service.ezid;
002
003import org.gbif.utils.file.FileUtils;
004
005import java.util.Map;
006
007import org.junit.Test;
008
009import static org.junit.Assert.assertEquals;
010
011public class AnvlUtilsTest {
012
013  @Test
014  public void testRead() throws Exception {
015    Map<String, String> map = AnvlUtils.read(FileUtils.classpathStream("anvl.txt"));
016    System.out.println(map);
017    assertEquals(16, map.size());
018    assertEquals("ark:/b6084/m9.figshare.821213", map.get("_shadowedby"));
019    assertEquals("http://figshare.com/articles/5000_random_journal_article_CrossRef_DOIs_from_2012/821213", map.get(AnvlUtils.TARGET));
020    assertEquals(869, map.get(AnvlUtils.DATACITE).length());
021    assertEquals("public", map.get(AnvlUtils.STATUS));
022  }
023
024  @Test
025  public void testRountrip() throws Exception {
026    Map<String, String> map = AnvlUtils.read(FileUtils.classpathStream("anvl.txt"));
027    String anvl = AnvlUtils.write(map);
028    assertEquals(map, AnvlUtils.read(anvl));
029  }
030}
