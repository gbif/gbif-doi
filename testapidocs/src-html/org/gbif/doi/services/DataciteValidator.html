001package org.gbif.doi.services;
002
003import java.io.IOException;
004import java.io.InputStream;
005import java.io.StringReader;
006import javax.xml.transform.stream.StreamSource;
007import javax.xml.validation.Schema;
008import javax.xml.validation.SchemaFactory;
009import javax.xml.validation.Validator;
010
011import org.xml.sax.SAXException;
012
013/**
014 * For the validation of DataCite metadata files.
015 */
016public class DataciteValidator {
017  private static Validator DATACITE_VALIDATOR;
018
019  /**
020   * @return an xml validator based on the DataCite metadata schema version 3
021   */
022  public static void validateMetadata(String xml) throws IOException, SAXException {
023    getValidator().validate(new StreamSource(new StringReader(xml)));
024  }
025
026  public static void validateMetadata(InputStream xml) throws IOException, SAXException {
027    getValidator().validate(new StreamSource(xml));
028  }
029
030  private static Validator getValidator() throws IOException, SAXException {
031    if (DATACITE_VALIDATOR == null) {
032      // define the type of schema - we use W3C:
033      String schemaLang = "http://www.w3.org/2001/XMLSchema";
034      // get validation driver:
035      SchemaFactory factory = SchemaFactory.newInstance(schemaLang);
036      // create schema by reading it from an URL:
037      //File f = FileUtils.getClasspathFile("datacite/metadata.xsd");
038      Schema schema = factory.newSchema(new StreamSource("http://schema.datacite.org/meta/kernel-3/metadata.xsd"));
039      DATACITE_VALIDATOR = schema.newValidator();
040    }
041    return DATACITE_VALIDATOR;
042  }
043
044}
