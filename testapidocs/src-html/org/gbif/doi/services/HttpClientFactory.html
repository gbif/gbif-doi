001package org.gbif.doi.services;
002
003import org.gbif.utils.HttpUtil;
004
005import java.util.concurrent.TimeUnit;
006
007import org.apache.http.impl.client.DefaultHttpClient;
008
009/**
010 * A factory that builds new Apache Http Clients to be used with the DOI service.
011 *
012 * This allows for easy swapping of implementations to make testing easier.
013 */
014public class HttpClientFactory {
015
016  private static final int MAX_TOTAL_CONNECTIONS = 200;
017  private static final int MAX_CONNECTIONS_PER_HOST = 20;
018
019  private final int timeout;
020
021  /**
022   * Builds a new Factory.
023   */
024  public HttpClientFactory(long timeout, TimeUnit timeUnit) {
025    long millis = TimeUnit.MILLISECONDS.convert(timeout, timeUnit);
026
027    System.setProperty("sun.net.client.defaultConnectTimeout", String.valueOf(millis));
028    System.setProperty("sun.net.client.defaultReadTimeout", String.valueOf(millis));
029
030    this.timeout = (int) millis;
031  }
032
033  public DefaultHttpClient provideHttpClient() {
034    return HttpUtil.newMultithreadedClient(timeout, MAX_TOTAL_CONNECTIONS, MAX_CONNECTIONS_PER_HOST);
035  }
036
037}
