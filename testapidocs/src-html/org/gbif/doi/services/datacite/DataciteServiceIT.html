001package org.gbif.doi.services.datacite;

012
013import java.io.IOException;
014import java.io.InputStream;
015import java.net.URI;
016import javax.xml.parsers.ParserConfigurationException;
017
018import com.fasterxml.jackson.databind.ObjectMapper;
019import com.fasterxml.jackson.dataformat.yaml.YAMLFactory;
020import org.junit.BeforeClass;
021import org.junit.Ignore;
022import org.junit.Test;
023import org.xml.sax.SAXException;
024
025import static org.junit.Assert.assertEquals;
026import static org.junit.Assert.fail;
027
028public class DataciteServiceIT {
029  // test prefix for DataCite, behaving like normal DOIs and occasionally purged
030  private static final String TEST_PREFIX = "10.5072";
031  private static final String GBIF_PREFIX = "10.15468";
032  private static DoiService service;
033
034  final static DOI FOREIGN_DOI = new DOI("10.6084/m9.figshare.821213");
035  final static URI TEST_TARGET = URI.create("http://www.gbif.org/datasets");
036
037  @BeforeClass
038  public static void setup() throws IOException, ParserConfigurationException, SAXException {
039    ObjectMapper mapper = new ObjectMapper(new YAMLFactory());
040    InputStream dc = FileUtils.classpathStream("datacite.yaml");
041    ServiceConfig cfg = mapper.readValue(dc, ServiceConfig.class);
042    System.out.println(cfg);
043    service = new DataciteService(HttpUtil.newMultithreadedClient(10000, 3, 2), cfg);
044  }
045
046  @Test
047  public void testGet() throws Exception {
048    final DOI doi = new DOI(TEST_PREFIX, "gbif-"+System.nanoTime());
049    DataCiteMetadata meta = DataCiteMetadataTest.testMetadata(doi, "reserve test");
050    service.reserve(doi, meta);
051    service.register(doi, TEST_TARGET);
052    assertEquals(TEST_TARGET, service.get(doi));
053  }
054
055  @Test(expected = DoiHttpException.class)
056  public void testGetForeign() throws Exception {
057    // this DOI does not belong to GBIF and we cannot read it...
058    service.get(FOREIGN_DOI);
059  }
060
061  @Test
062  public void testReserve() throws Exception {
063    final DOI doi = new DOI(TEST_PREFIX, "gbif-"+System.nanoTime());
064    DataCiteMetadata meta = DataCiteMetadataTest.testMetadata(doi, "reserve test");
065    service.reserve(doi, meta);
066    try {
067      service.reserve(doi, meta);
068      fail("DOI was reserved already, expected DoiExistsException");
069    } catch (DoiException e) {
070
071    }
072  }
073
074  @Test
075  public void testReserveRandom() throws Exception {
076    DataCiteMetadata meta = DataCiteMetadataTest.testMetadata();
077    DOI doi = service.reserveRandom(TEST_PREFIX, 4, meta);
078    System.out.println("Reserved random id " + doi);
079  }
080
081  @Test
082  public void testDeleteReserved() throws Exception {
083    final DOI doi = new DOI(TEST_PREFIX, "gbif-"+System.nanoTime());
084    DataCiteMetadata meta = DataCiteMetadataTest.testMetadata(doi, "reserve test");
085    service.reserve(doi, meta);
086    // now delete it and reserving again should work
087    service.delete(doi);
088    service.reserve(doi, meta);
089    service.delete(doi);
090  }
091
092  @Test
093  public void testDeleteRegistered() throws Exception {
094    final DOI doi = new DOI(TEST_PREFIX, "gbif-"+System.nanoTime());
095    DataCiteMetadata meta = DataCiteMetadataTest.testMetadata(doi, "reserve test");
096    service.reserve(doi, meta);
097    service.register(doi, TEST_TARGET);
098    assertEquals(TEST_TARGET, service.get(doi));
099    // now delete it to remove metadata, the registered target remains
100    service.delete(doi);
101    assertEquals(TEST_TARGET, service.get(doi));
102    // reserving again should work
103    service.reserve(doi, meta);
104  }
105
106  @Test
107  public void testUpdateRegistered() throws Exception {
108    final DOI doi = new DOI(TEST_PREFIX, "gbif-"+System.nanoTime());
109    DataCiteMetadata meta = DataCiteMetadataTest.testMetadata(doi, "reserve test");
110    service.reserve(doi, meta);
111    service.register(doi, TEST_TARGET);
112    assertEquals(TEST_TARGET, service.get(doi));
113
114    final URI target2 = TEST_TARGET.resolve("subsub");
115    service.update(doi, target2);
116    assertEquals(target2, service.get(doi));
117  }
118
119  @Test
120  public void testRegister() throws Exception {
121    final DOI doi = new DOI(TEST_PREFIX, "gbif-"+System.nanoTime());
122    DataCiteMetadata meta = DataCiteMetadataTest.testMetadata(doi, "reserve test");
123    service.reserve(doi, meta);
124    service.register(doi, TEST_TARGET);
125    assertEquals(TEST_TARGET, service.get(doi));
126    try {
127      // try again, this should fail!
128      service.register(doi, TEST_TARGET);
129      fail("DOI was registered already, expected DoiExistsException");
130    } catch (DoiException e) {
131      System.out.println(e);
132    }
133  }
134
135  @Test(expected = DoiException.class)
136  public void testRegisterFail() throws Exception {
137    final DOI doi = new DOI(TEST_PREFIX, "gbif-"+System.nanoTime());
138    // registering without previous reservation should throw an error
139    service.register(doi, TEST_TARGET);
140  }

142  @Test
143  @Ignore("This is only to manually test the real prefix behavior. Only use the test prefix for regular tests")
144  public void testRealGBIFPrefix() throws Exception {
145
146    final DOI doi = new DOI(GBIF_PREFIX, "test."+System.nanoTime());
147    System.out.println(doi);
148    URI url = service.get(doi);
149    System.out.println(url);
150
151    DataCiteMetadata meta = DataCiteMetadataTest.testMetadata(doi, "reserve test");
152    service.reserve(doi, meta);
153    url = service.get(doi);
154    System.out.println(url);
155
156    service.register(doi, TEST_TARGET);
157    url = service.get(doi);

159  }
