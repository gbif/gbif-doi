016package org.gbif.doi.util;
017
018import org.gbif.doi.metadata.datacite.DataCiteMetadata;
019import org.gbif.doi.service.datacite.DataCiteValidator;
020import org.gbif.doi.util.Difference.DifferenceItem;
021
022import java.io.IOException;
023import java.nio.file.Files;
024import java.nio.file.Paths;
025import java.util.List;
026import java.util.Objects;
027import java.util.stream.Collectors;
028
029import org.jetbrains.annotations.NotNull;
030import org.junit.BeforeClass;
031import org.junit.Test;
032
033import static org.junit.Assert.assertEquals;
034import static org.junit.Assert.assertFalse;
035import static org.junit.Assert.assertTrue;
036
037@SuppressWarnings("ConstantConditions")
038public class MetadataUtilsTest {
039
040  private static String metadataXmlLowerCaseIdentifier;
041  private static String metadataXmlLowerCaseIdentifierAnother;
042  private static String metadataXmlUpperCaseIdentifier;
043
044  @BeforeClass
045  public static void before() throws Exception {
046    metadataXmlLowerCaseIdentifier = readFileData("metadata/datacite-example-full-v4.xml");
047    metadataXmlUpperCaseIdentifier =
048        readFileData("metadata/datacite-example-full-v4-uppercase-identifier.xml");
049    metadataXmlLowerCaseIdentifierAnother =
050        readFileData("metadata/datacite-example-full-v4-another.xml");
051  }
052
053  @NotNull
054  private static String readFileData(String s) throws IOException {
055    return new String(
056        Files.readAllBytes(Paths.get(ClassLoader.getSystemClassLoader().getResource(s).getFile())));
057  }
058
059  @Test
060  public void testEqualInsensitiveIdentifierDataCiteMetadataSources() throws Exception {
061    // given
062    DataCiteMetadata uppercaseIdentifierMetadata =
063        DataCiteValidator.fromXml(metadataXmlLowerCaseIdentifier);
064    DataCiteMetadata lowercaseIdentifierMetadata =
065        DataCiteValidator.fromXml(metadataXmlUpperCaseIdentifier);
066
067    // when
068    boolean metadataEqualsResult =
069        MetadataUtils.metadataEquals(uppercaseIdentifierMetadata, lowercaseIdentifierMetadata);
070    boolean regularEqualsResult =
071        Objects.equals(uppercaseIdentifierMetadata, lowercaseIdentifierMetadata);
072
073    // then
074    assertTrue(metadataEqualsResult);
075    assertFalse(regularEqualsResult);
076  }
077
078  @Test
079  public void testEqualInsensitiveIdentifierStringSources() {
080    // when
081    boolean metadataEqualsResult =
082        MetadataUtils.metadataEquals(
083            metadataXmlUpperCaseIdentifier, metadataXmlLowerCaseIdentifier);
084    boolean regularEqualsResult =
085        Objects.equals(metadataXmlUpperCaseIdentifier, metadataXmlLowerCaseIdentifier);
086
087    // then
088    assertTrue(metadataEqualsResult);
089    assertFalse(regularEqualsResult);
090  }
091
092  @Test
093  public void testMetadataDifference() {
094    // when
095    Difference metadataDifference =
096        MetadataUtils.metadataDifference(
097            metadataXmlLowerCaseIdentifier, metadataXmlLowerCaseIdentifierAnother);
098
099    // then
100    List<String> fields =
101        metadataDifference.getItems().stream()
102            .map(DifferenceItem::getFieldName)
103            .collect(Collectors.toList());
104    assertEquals(3, fields.size());
105    assertTrue(fields.contains("creators"));
106    assertTrue(fields.contains("publicationYear"));
107    assertTrue(fields.contains("dates"));
108  }
109
110  @Test
111  public void testMetadataDifferenceBothArgumentNull() {
112    // when
113    Difference metadataDifference = MetadataUtils.metadataDifference(null, null);
114
115    // then
116    assertEquals(0, metadataDifference.getItems().size());
117  }
118
119  @Test
120  public void testMetadataDifferenceOneArgumentNull() {
121    // when
122    Difference metadataDifference =
123        MetadataUtils.metadataDifference(null, metadataXmlLowerCaseIdentifier);
124
125    // then
126    assertEquals(0, metadataDifference.getItems().size());
127  }
128}
