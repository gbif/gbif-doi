016package org.gbif.doi.util;
017
018import org.gbif.doi.metadata.datacite.DataCiteMetadata;
019import org.gbif.doi.service.datacite.DataCiteValidator;
020import org.gbif.doi.util.Difference.DifferenceItem;
021
022import java.io.IOException;
023import java.nio.file.Files;
024import java.nio.file.Paths;
025import java.util.List;
026import java.util.Objects;
027import java.util.stream.Collectors;
028
029import org.junit.jupiter.api.BeforeAll;
030import org.junit.jupiter.api.Test;
031
032import static org.junit.jupiter.api.Assertions.assertEquals;
033import static org.junit.jupiter.api.Assertions.assertFalse;
034import static org.junit.jupiter.api.Assertions.assertTrue;
035
036@SuppressWarnings("ConstantConditions")
037public class MetadataUtilsTest {
038
039  private static String metadataXmlLowerCaseIdentifier;
040  private static String metadataXmlLowerCaseIdentifierAnother;
041  private static String metadataXmlUpperCaseIdentifier;
042  private static String metadataXmlVersion3;
043
044  @BeforeAll
045  public static void before() throws Exception {
046    metadataXmlLowerCaseIdentifier = readFileData("metadata/datacite-example-full-v4.xml");
047    metadataXmlUpperCaseIdentifier =
048        readFileData("metadata/datacite-example-full-v4-uppercase-identifier.xml");
049    metadataXmlLowerCaseIdentifierAnother =
050        readFileData("metadata/datacite-example-full-v4-another.xml");
051    metadataXmlVersion3 = readFileData("metadata/datacite-example-full-v3.xml");
052  }
053
054  private static String readFileData(String s) throws IOException {
055    return new String(
056        Files.readAllBytes(Paths.get(ClassLoader.getSystemClassLoader().getResource(s).getFile())));
057  }
058
059  @Test
060  public void testEqualInsensitiveIdentifierDataCiteMetadataSources() throws Exception {
061    // given
062    DataCiteMetadata uppercaseIdentifierMetadata =
063        DataCiteValidator.fromXml(metadataXmlLowerCaseIdentifier);
064    DataCiteMetadata lowercaseIdentifierMetadata =
065        DataCiteValidator.fromXml(metadataXmlUpperCaseIdentifier);
066
067    // when
068    boolean metadataEqualsResult =
069        MetadataUtils.metadataEquals(uppercaseIdentifierMetadata, lowercaseIdentifierMetadata);
070    boolean regularEqualsResult =
071        Objects.equals(uppercaseIdentifierMetadata, lowercaseIdentifierMetadata);
072
073    // then
074    assertTrue(metadataEqualsResult);
075    assertFalse(regularEqualsResult);
076  }
077
078  @Test
079  public void testEqualInsensitiveIdentifierStringSources() {
080    // when
081    boolean metadataEqualsResult =
082        MetadataUtils.metadataEquals(
083            metadataXmlUpperCaseIdentifier, metadataXmlLowerCaseIdentifier);
084    boolean regularEqualsResult =
085        Objects.equals(metadataXmlUpperCaseIdentifier, metadataXmlLowerCaseIdentifier);
086
087    // then
088    assertTrue(metadataEqualsResult);
089    assertFalse(regularEqualsResult);
090  }
091
092  @Test
093  public void testEqualOneOrBothMetadataNull() {
094    // when
095    boolean firstMetadataNullResult =
096        MetadataUtils.metadataEquals(null, metadataXmlUpperCaseIdentifier);
097    boolean secondMetadataNullResult =
098        MetadataUtils.metadataEquals(metadataXmlUpperCaseIdentifier, null);
099    boolean bothMetadataNullResult = MetadataUtils.metadataEquals((String) null, null);
100
101    // then
102    assertFalse(firstMetadataNullResult);
103    assertFalse(secondMetadataNullResult);
104    assertTrue(bothMetadataNullResult);
105  }
106
107  @Test
108  public void testEqualAllowOnlyRecentMetadata() {
109    // when
110    boolean result = MetadataUtils.metadataEquals(metadataXmlVersion3, metadataXmlVersion3);
111
112    // then
113    assertFalse(result);
114  }
115
116  @Test
117  public void testMetadataDifference() {
118    // when
119    Difference metadataDifference =
120        MetadataUtils.metadataDifference(
121            metadataXmlLowerCaseIdentifier, metadataXmlLowerCaseIdentifierAnother);
122
123    // then
124    List<String> fields =
125        metadataDifference.getItems().stream()
126            .map(DifferenceItem::getFieldName)
127            .collect(Collectors.toList());
128    assertEquals(3, fields.size());
129    assertTrue(fields.contains("creators"));
130    assertTrue(fields.contains("publicationYear"));
131    assertTrue(fields.contains("dates"));
132    assertEquals(Difference.Status.COMPATIBLE_OBJECTS, metadataDifference.getComparisonStatus());
133  }
134
135  @Test
136  public void testMetadataDifferenceOneOrBothArgumentsNull() {
137    // when
138    Difference metadataDifferenceFirstArgumentNull =
139        MetadataUtils.metadataDifference(null, metadataXmlLowerCaseIdentifier);
140    Difference metadataDifferenceSecondArgumentNull =
141        MetadataUtils.metadataDifference(metadataXmlLowerCaseIdentifier, null);
142    Difference metadataDifferenceBothArgumentNull = MetadataUtils.metadataDifference(null, null);
143
144    // then
145    assertEquals(0, metadataDifferenceFirstArgumentNull.getItems().size());
146    assertEquals(0, metadataDifferenceSecondArgumentNull.getItems().size());
147    assertEquals(0, metadataDifferenceBothArgumentNull.getItems().size());
148    assertEquals(
149        Difference.Status.NULL_OBJECTS, metadataDifferenceFirstArgumentNull.getComparisonStatus());
150    assertEquals(
151        Difference.Status.NULL_OBJECTS, metadataDifferenceSecondArgumentNull.getComparisonStatus());
152    assertEquals(
153        Difference.Status.NULL_OBJECTS, metadataDifferenceBothArgumentNull.getComparisonStatus());
154  }
155
156  @Test
157  public void testMetadataDifferenceIncompatibleMetadataVersion() {
158    // when
159    Difference metadataDifferenceFirstArgumentIncompatibleMetadata =
160        MetadataUtils.metadataDifference(metadataXmlLowerCaseIdentifier, metadataXmlVersion3);
161    Difference metadataDifferenceSecondArgumentIncompatibleMetdata =
162        MetadataUtils.metadataDifference(metadataXmlVersion3, metadataXmlLowerCaseIdentifier);
163    Difference metadataDifferenceBothArgumentIncompatibleMetadata =
164        MetadataUtils.metadataDifference(metadataXmlVersion3, metadataXmlVersion3);
165
166    // then
167    assertEquals(0, metadataDifferenceFirstArgumentIncompatibleMetadata.getItems().size());
168    assertEquals(0, metadataDifferenceSecondArgumentIncompatibleMetdata.getItems().size());
169    assertEquals(0, metadataDifferenceBothArgumentIncompatibleMetadata.getItems().size());
170    assertEquals(
171        Difference.Status.INCOMPATIBLE_OBJECTS,
172        metadataDifferenceFirstArgumentIncompatibleMetadata.getComparisonStatus());
173    assertEquals(
174        Difference.Status.INCOMPATIBLE_OBJECTS,
175        metadataDifferenceSecondArgumentIncompatibleMetdata.getComparisonStatus());
176    assertEquals(
177        Difference.Status.INCOMPATIBLE_OBJECTS,
178        metadataDifferenceBothArgumentIncompatibleMetadata.getComparisonStatus());
179  }
