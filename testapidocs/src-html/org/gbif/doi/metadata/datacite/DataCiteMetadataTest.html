228  /** Just check everything was generated properly. */
229  @Test
230  public void testObjectFactory() {
231    ObjectFactory of = new ObjectFactory();
232    DataCiteMetadata res = of.createDataCiteMetadata();
233
234    Creators creators = of.createDataCiteMetadataCreators();
235    Creator creator = of.createDataCiteMetadataCreatorsCreator();
236    CreatorName cn = of.createDataCiteMetadataCreatorsCreatorCreatorName();
237    cn.setValue("Piyapong");
238    creator.setCreatorName(cn);
239
240    NameIdentifier nid = of.createNameIdentifier();
241    nid.setValue("pc1405");
242    creator.getNameIdentifier().add(nid);
243    creators.getCreator().add(creator);
244    res.setCreators(creators);
245
246    Identifier id = of.createDataCiteMetadataIdentifier();
247    id.setIdentifierType("DOI");
248    id.setValue("10.21373/WDCC/CCSRNIES_SRES_B2");
249    res.setIdentifier(id);
250
251    Titles titles = of.createDataCiteMetadataTitles();
252    Title title = of.createDataCiteMetadataTitlesTitle();
253    title.setValue(
254        "National Institute for Environmental Studies and Center for Climate System Research Japan");
255    titles.getTitle().add(title);
256    res.setTitles(titles);
257
258    res.setPublicationYear("2014");
259    Publisher pub = of.createDataCiteMetadataPublisher();
260    pub.setValue("Piyapong");
261    res.setPublisher(pub);
262
263    assertNotNull(res);
264    System.out.print(res);
265  }
