008public class DataCiteMetadataTest {
009
010  public static DataCiteMetadata testMetadata() {
011    return testMetadata(new DOI("10.5072/WDCC/CCSRNIES_SRES_B2"), "My Title");
012  }
013  public static DataCiteMetadata testMetadata(DOI doi, String title) {
014    return DataCiteMetadata.builder()
015      .withIdentifier().withValue(doi.getDoiName()).withIdentifierType("DOI").end()
016      .withResourceType(DataCiteMetadata.ResourceType.builder().withResourceTypeGeneral(ResourceType.DATASET).build())
017      .withCreators()
018      .addCreator().withCreatorName("Markus")
019      .withNameIdentifier().withNameIdentifierScheme("ORCID").withSchemeURI("orcid.org").withValue("0000-0001-7757-1889").end()
020      .end()
021      .end()
022      .withTitles()
023      .addTitle().withValue(title).withLang(Language.ENGLISH.getIso3LetterCode()).end()
024      .addTitle().withValue("Meine Ãœbersetzung").withLang(Language.GERMAN.getIso3LetterCode()).withTitleType(
025        TitleType.TRANSLATED_TITLE).end()
026      .end()
027      .withPublicationYear("2014")
028      .withPublisher("Pensoft")
029      .build();
030  }
031
032  @Test
033  public void testObjectFactory() throws Exception {
034    ObjectFactory of = new ObjectFactory();
035    DataCiteMetadata res = of.createDataCiteMetadata();
036
037    DataCiteMetadata.Creators creators = of.createDataCiteMetadataCreators();
038    DataCiteMetadata.Creators.Creator creator = of.createDataCiteMetadataCreatorsCreator();
039    creator.setCreatorName("Piyapong");
040    DataCiteMetadata.Creators.Creator.NameIdentifier nid = of.createDataCiteMetadataCreatorsCreatorNameIdentifier();
041    nid.setValue("pc1405");
042    creator.setNameIdentifier(nid);
043    creators.getCreator().add(creator);
044    res.setCreators(creators);
045
046    DataCiteMetadata.Identifier id = of.createDataCiteMetadataIdentifier();
047    id.setIdentifierType("DOI");
048    id.setValue("10.5072/WDCC/CCSRNIES_SRES_B2");
049    res.setIdentifier(id);
050
051    DataCiteMetadata.Titles titles = of.createDataCiteMetadataTitles();
052    DataCiteMetadata.Titles.Title title = of.createDataCiteMetadataTitlesTitle();
053    title.setValue("National Institute for Environmental Studies and Center for Climate System Research Japan");
054    titles.getTitle().add(title);
055    res.setTitles(titles);
056
057    res.setPublicationYear("2014");
058    res.setPublisher("Piyapong");
059
060    System.out.print(res);
061  }
062
