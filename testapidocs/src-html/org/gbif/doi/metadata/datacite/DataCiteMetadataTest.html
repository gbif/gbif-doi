001package org.gbif.doi.metadata.datacite;
002
003import org.gbif.api.vocabulary.Language;
004
005import org.junit.Test;
006
007public class DataCiteMetadataTest {
008  
009  @Test
010  public void testObjectFactory() throws Exception {
011    ObjectFactory of = new ObjectFactory();
012    DataCiteMetadata res = of.createDataCiteMetadata();
013
014    DataCiteMetadata.Creators creators = of.createDataCiteMetadataCreators();
015    DataCiteMetadata.Creators.Creator creator = of.createDataCiteMetadataCreatorsCreator();
016    creator.setCreatorName("Piyapong");
017    DataCiteMetadata.Creators.Creator.NameIdentifier nid = of.createDataCiteMetadataCreatorsCreatorNameIdentifier();
018    nid.setValue("pc1405");
019    creator.setNameIdentifier(nid);
020    creators.getCreator().add(creator);
021    res.setCreators(creators);
022
023    DataCiteMetadata.Identifier id = of.createDataCiteMetadataIdentifier();
024    id.setIdentifierType("DOI");
025    id.setValue("10.5072/WDCC/CCSRNIES_SRES_B2");
026    res.setIdentifier(id);
027
028    DataCiteMetadata.Titles titles = of.createDataCiteMetadataTitles();
029    DataCiteMetadata.Titles.Title title = of.createDataCiteMetadataTitlesTitle();
030    title.setValue("National Institute for Environmental Studies and Center for Climate System Research Japan");
031    titles.getTitle().add(title);
032    res.setTitles(titles);
033
034    res.setPublicationYear("2014");
035    res.setPublisher("Piyapong");
036
037    System.out.print(res);
038  }
039
040  @Test
041  public void testBuilder() throws Exception {
042    DataCiteMetadata r = DataCiteMetadata.builder()
043      .withResourceType(DataCiteMetadata.ResourceType.builder().withResourceTypeGeneral(ResourceType.DATASET).build())
044      .withCreators()
045      .addCreator().withCreatorName("Markus")
046      .withNameIdentifier().withNameIdentifierScheme("ORCID").withSchemeURI("orcid.org").withValue("12445322").end()
047      .end()
048      .end()
049      .withTitles()
050      .addTitle().withValue("My Title").withLang(Language.GERMAN.getIso3LetterCode()).withTitleType(
051        TitleType.TRANSLATED_TITLE).end()
052      .end()
053      .build();
054    System.out.print(r);
055  }
056}
