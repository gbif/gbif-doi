020  @Test
021  public void testClientRedirect() throws ParseException, IOException {
022    HttpUtil util = new HttpUtil(new DefaultHttpClient());
023    File tmp = File.createTempFile("httputils", "test");
024    tmp.deleteOnExit();
025    // a redirect to http://rs.gbif.org/extension/gbif/1.0/distribution.xml
026    StatusLine status = util.download("http://rs.gbif.org/terms/1.0/Distribution", tmp);
027    assertTrue(HttpUtil.success(status));
028
029    // assert false non 404s
030    status = util.download("http://rs.gbif.org/hoppladi/hopplaho/hallodrian/tim/is/back.txt", tmp);
031    assertFalse(HttpUtil.success(status));
032    assertEquals(404, status.getStatusCode());
033
034  }

036  /**
037   * Tests a condition get against an apache http server within GBIF.
038   * If the file being tested against has been changed in the last 24h this test is expected to fail!
039   * 
040   * @see <a href="http://dev.gbif.org/issues/browse/GBIFCOM-77">GBIFCOM-77</a>
041   */
042  @Test
043  public void testConditionalGet() throws ParseException, IOException {
044    DefaultHttpClient client = new DefaultHttpClient();
045    HttpUtil util = new HttpUtil(client);
046    // We know for sure it has changed since this date
047    Date last = HttpUtil.parseHeaderDate("Wed, 03 Aug 2009 22:37:31 GMT");
048    File tmp = File.createTempFile("vocab", ".xml");
049    URL url = new URL("http://rs.gbif.org/vocabulary/gbif/rank.xml");
050    assertTrue(util.downloadIfChanged(url, last, tmp));
051
052    // Verify that it does not download with a conditional get of a day before "now"
053    // we dont use now because the apache server on the other side might have a slightly different clock and timezone
054    // configured that might lead to a mismatch.
055    // If the rank file was modified within the last 24h this test could fail though !!!
056    Date nearlyNow = new Date(System.currentTimeMillis() - 24 * 1000 * 60 * 60);
057    assertFalse(util.downloadIfChanged(url, nearlyNow, tmp));
058  }
